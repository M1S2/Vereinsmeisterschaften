81 public async Task<bool> Load(string path, CancellationToken cancellationToken)
82 {
83 bool importingResult = false;
84 Exception exception = null;
85 await Task.Run(() =>
86 {
87 try
88 {
89 if (!File.Exists(path))
90 {
91 OnFileProgress?.Invoke(this, 0);
92 _competitionList.Clear();
93 OnFileProgress?.Invoke(this, 100);
94 }
95 else
96 {
97 List<Competition> list = _fileService.LoadFromCsv<Competition>(path, cancellationToken, Competition.SetPropertyFromString, OnFileProgress);
98 _competitionList = new ObservableCollection<Competition>();
99 foreach (Competition competition in list)
100 {
101 AddCompetition(competition);
102 }
103 }
104
105 _competitionListOnLoad = _competitionList.ToList().ConvertAll(c => new Competition(c));
106
107 PersistentPath = path;
108 importingResult = true;
109 }
110 catch (OperationCanceledException)
111 {
112 importingResult = false;
113 }
114 catch (Exception ex)
115 {
116 exception = ex;
117 }
118 });
119 OnFileFinished?.Invoke(this, null);
120 if (exception != null) { throw exception; }
121 return importingResult;
122 }

254 public Competition GetCompetitionForPerson(Person person, SwimmingStyles swimmingStyle, out bool isUsingMaxAgeCompetition)
255 {
256 // Find the competition with the maximum age that matches the swimming style and gender
257 byte maxAge = _competitionList.Where(c => c.Gender == person.Gender && c.SwimmingStyle == swimmingStyle).OrderByDescending(c => c.Age).FirstOrDefault()?.Age ?? 0;
258
259 ushort competitionYear = _workspaceService?.Settings?.GetSettingValue<ushort>(WorkspaceSettings.GROUP_GENERAL, WorkspaceSettings.SETTING_GENERAL_COMPETITIONYEAR) ?? 0;
260 int personAge = competitionYear - person.BirthYear;
261 Competition foundCompetition = _competitionList.Where(c => c.Gender == person.Gender &&
262 c.SwimmingStyle == swimmingStyle &&
263 (c.SwimmingStyle == SwimmingStyles.WaterFlea ? personAge <= c.Age : c.Age == (personAge <= maxAge ? personAge : maxAge))).FirstOrDefault();
264
265 isUsingMaxAgeCompetition = foundCompetition != null && foundCompetition.Age == maxAge && personAge > maxAge && foundCompetition.SwimmingStyle != SwimmingStyles.WaterFlea;
266 return foundCompetition;
267 }

275 public void UpdateAllCompetitionsForPerson(Person person)
276 {
277 // Update the available competitions for the person
278 List<SwimmingStyles> _availableSwimmingStyles = Enum.GetValues(typeof(SwimmingStyles)).Cast<SwimmingStyles>().Where(s => s != SwimmingStyles.Unknown).ToList();
279 Dictionary<SwimmingStyles, Competition> availableCompetitions = new Dictionary<SwimmingStyles, Competition>();
280 Dictionary<SwimmingStyles, bool> isUsingMaxAgeCompetitionDict = new Dictionary<SwimmingStyles, bool>();
281 foreach (SwimmingStyles swimmingStyle in _availableSwimmingStyles)
282 {
283 bool isUsingMaxAgeCompetition;
284 Competition competition = GetCompetitionForPerson(person, swimmingStyle, out isUsingMaxAgeCompetition);
285 if (availableCompetitions.ContainsKey(swimmingStyle))
286 {
287 availableCompetitions[swimmingStyle] = competition;
288 }
289 else
290 {
291 availableCompetitions.Add(swimmingStyle, competition);
292 }
293
294 if (isUsingMaxAgeCompetitionDict.ContainsKey(swimmingStyle))
295 {
296 isUsingMaxAgeCompetitionDict[swimmingStyle] = isUsingMaxAgeCompetition;
297 }
298 else
299 {
300 isUsingMaxAgeCompetitionDict.Add(swimmingStyle, isUsingMaxAgeCompetition);
301 }
302 }
303 person.AvailableCompetitions = availableCompetitions;
304 person.IsUsingMaxAgeCompetitionDict = isUsingMaxAgeCompetitionDict;
305
306 // Update the competitions for the person starts
307 foreach (PersonStart personStart in _personService.GetAllPersonStarts(PersonStartFilters.Person, person))
308 {
309 personStart.CompetitionObj = person.AvailableCompetitions[personStart.Style];
310 personStart.IsUsingMaxAgeCompetition = person.IsUsingMaxAgeCompetitionDict[personStart.Style];
311 }
312 }
