94 public async Task<bool> Load(string path, CancellationToken cancellationToken)
95 {
96 bool importingResult = false;
97 Exception exception = null;
98 await Task.Run(() =>
99 {
100 try
101 {
102 if (!File.Exists(path))
103 {
104 OnFileProgress?.Invoke(this, 0);
105 _competitionList.Clear();
106 OnFileProgress?.Invoke(this, 100);
107 }
108 else
109 {
110 List<Competition> list = _fileService.LoadFromCsv<Competition>(path, cancellationToken, Competition.SetPropertyFromString, OnFileProgress, (header) =>
111 {
112 return PropertyNameLocalizedStringHelper.FindProperty(typeof(Competition), header);
113 });
114 _competitionList = new ObservableCollection<Competition>();
115 foreach (Competition competition in list)
116 {
117 AddCompetition(competition);
118 }
119 }
120
121 _competitionListOnLoad = _competitionList.ToList().ConvertAll(c => new Competition(c));
122
123 PersistentPath = path;
124 importingResult = true;
125 }
126 catch (OperationCanceledException)
127 {
128 importingResult = false;
129 }
130 catch (Exception ex)
131 {
132 exception = ex;
133 }
134 });
135 OnFileFinished?.Invoke(this, null);
136 if (exception != null) { throw exception; }
137 return importingResult;
138 }

175 saveResult = true;

289 public Competition GetCompetitionForPerson(Person person, SwimmingStyles swimmingStyle, out bool isUsingMaxAgeCompetition, out bool isUsingExactAgeCompetition)
290 {
291 isUsingMaxAgeCompetition = false;
292 isUsingExactAgeCompetition = true;
293
294 // Get all competitions that match the gender and swimming style
295 List<Competition> competitions = _competitionList.Where(c => c.Gender == person.Gender && c.SwimmingStyle == swimmingStyle)
296 .OrderBy(c => c.Age).ToList();
297 if (competitions.Count == 0) { return null; }
298
299 CompetitionSearchModes searchMode = _workspaceService?.Settings?.GetSettingValue<CompetitionSearchModes>(WorkspaceSettings.GROUP_GENERAL, WorkspaceSettings.SETTING_GENERAL_COMPETITIONSEARCHMODE) ?? CompetitionSearchModes.ExactOrNextLowerOnlyMaxAge;
300 ushort competitionYear = _workspaceService?.Settings?.GetSettingValue<ushort>(WorkspaceSettings.GROUP_GENERAL, WorkspaceSettings.SETTING_GENERAL_COMPETITIONYEAR) ?? 0;
301 int personAge = competitionYear - person.BirthYear;
302 byte maxAge = competitions.Max(c => c.Age);
303 byte minAge = competitions.Min(c => c.Age);
304
305 // Special case WaterFlea
306 if (swimmingStyle == SwimmingStyles.WaterFlea)
307 {
308 return competitions.Where(c => personAge <= c.Age).FirstOrDefault();
309 }
310
311 Competition foundCompetition = null;
312 switch (searchMode)
313 {
314 case CompetitionSearchModes.OnlyExactAge:
315 foundCompetition = competitions.FirstOrDefault(c => c.Age == personAge);
316 break;
317 case CompetitionSearchModes.ExactOrNextLowerAge:
318 foundCompetition = competitions.LastOrDefault(c => c.Age <= personAge);
319 break;
320 case CompetitionSearchModes.ExactOrNextHigherAge:
321 foundCompetition = competitions.FirstOrDefault(c => c.Age >= personAge);
322 break;
323 case CompetitionSearchModes.ExactOrNextLowerOnlyMaxAge:
324 foundCompetition = competitions.FirstOrDefault(c => c.Age == personAge);
325 if (foundCompetition == null && personAge > maxAge) // if no exact match was found and the person age is greater than the max age, use the competition with the max age
326 {
327 foundCompetition = competitions.FirstOrDefault(c => c.Age == maxAge);
328 }
329 break;
330 case CompetitionSearchModes.ExactOrNearestPreferLower:
331 foundCompetition = competitions.FirstOrDefault(c => c.Age == personAge);
332 if (foundCompetition == null && personAge >= minAge)
333 {
334 // Calculate age distance and order by minimal distance. Prefer lower age on equal distance.
335 foundCompetition = competitions.OrderBy(c => Math.Abs(c.Age - personAge)).ThenBy(c => c.Age).FirstOrDefault();
336 }
337 break;
338 case CompetitionSearchModes.ExactOrNearestPreferHigher:
339 foundCompetition = competitions.FirstOrDefault(c => c.Age == personAge);
340 if (foundCompetition == null && personAge >= minAge)
341 {
342 // Calculate age distance and order by minimal distance. Prefer higher age on equal distance.
343 foundCompetition = competitions.OrderBy(c => Math.Abs(c.Age - personAge)).ThenByDescending(c => c.Age).FirstOrDefault();
344 }
345 break;
346 }
347
348 if (foundCompetition != null && foundCompetition.Age == maxAge) { isUsingMaxAgeCompetition = true; }
349 if (foundCompetition != null && foundCompetition.Age != personAge) { isUsingExactAgeCompetition = false; }
350
351 return foundCompetition;
352 }

360 public void UpdateAllCompetitionsForPerson(Person person)
361 {
362 // Update the available competitions for the person
363 List<SwimmingStyles> _availableSwimmingStyles = Enum.GetValues(typeof(SwimmingStyles)).Cast<SwimmingStyles>().Where(s => s != SwimmingStyles.Unknown).ToList();
364 Dictionary<SwimmingStyles, Competition> availableCompetitions = new Dictionary<SwimmingStyles, Competition>();
365 Dictionary<SwimmingStyles, bool> isUsingMaxAgeCompetitionDict = new Dictionary<SwimmingStyles, bool>();
366 Dictionary<SwimmingStyles, bool> isUsingExactAgeCompetitionDict = new Dictionary<SwimmingStyles, bool>();
367 foreach (SwimmingStyles swimmingStyle in _availableSwimmingStyles)
368 {
369 bool isUsingMaxAgeCompetition, isUsingExactAgeCompetition;
370 Competition competition = GetCompetitionForPerson(person, swimmingStyle, out isUsingMaxAgeCompetition, out isUsingExactAgeCompetition);
371
372 availableCompetitions[swimmingStyle] = competition;
373 isUsingMaxAgeCompetitionDict[swimmingStyle] = isUsingMaxAgeCompetition;
374 isUsingExactAgeCompetitionDict[swimmingStyle] = isUsingExactAgeCompetition;
375 }
376 person.AvailableCompetitions = availableCompetitions;
377 person.IsUsingMaxAgeCompetitionDict = isUsingMaxAgeCompetitionDict;
378 person.IsUsingExactAgeCompetitionDict = isUsingExactAgeCompetitionDict;
379
380 // Update the competitions for the person starts
381 foreach (PersonStart personStart in _personService.GetAllPersonStarts(PersonStartFilters.Person, person))
382 {
383 personStart.CompetitionObj = person.AvailableCompetitions[personStart.Style];
384 personStart.IsUsingMaxAgeCompetition = person.IsUsingMaxAgeCompetitionDict[personStart.Style];
385 personStart.IsUsingExactAgeCompetition = person.IsUsingExactAgeCompetitionDict[personStart.Style];
386 }
387 }
