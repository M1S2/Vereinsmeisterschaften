168 private void initPlaceholderViewConfigs()
169 {
170 PlaceholderViewConfigs.Clear();
171 foreach(KeyValuePair<string, List<string>> placeholders in Placeholders.PlaceholderDict)
172 {
173 string placeholderKey = placeholders.Key;
174 ResourceManager rm = new ResourceManager(typeof(Properties.PlaceholderResources));
175 string resourceStringName = rm.GetString($"PlaceholderName{placeholderKey}") ?? "?";
176 string resourceStringInfo = rm.GetString($"PlaceholderInfo{placeholderKey}") ?? "?";
177
178 Dictionary<DocumentCreationTypes, bool> isSupportedForDocumentType = new Dictionary<DocumentCreationTypes, bool>();
179 Dictionary<DocumentCreationTypes, string> postfixNumbersSupportedForDocumentType = new Dictionary<DocumentCreationTypes, string>();
180 foreach (IDocumentStrategy strategy in _documentStrategies)
181 {
182 DocumentCreationTypes documentCreationType = strategy.DocumentType;
183 bool isCurrentPlaceholderSupportedForStrategy = strategy.SupportedPlaceholderKeys.Contains(placeholderKey);
184
185 int indexOfKey = strategy.SupportedPlaceholderKeys.IndexOf(placeholderKey);
186 int postfixNumbersSupported = (indexOfKey >= 0 && indexOfKey < strategy.PostfixNumbersSupported.Count) ? strategy.PostfixNumbersSupported[indexOfKey] : 0;
187 string postfixNumbersSupportedStr = postfixNumbersSupported > 0 ? postfixNumbersSupported.ToString() : string.Empty;
188
189 if (!isSupportedForDocumentType.ContainsKey(documentCreationType))
190 {
191 isSupportedForDocumentType.Add(documentCreationType, isCurrentPlaceholderSupportedForStrategy);
192 postfixNumbersSupportedForDocumentType.Add(documentCreationType, postfixNumbersSupportedStr);
193 }
194 else
195 {
196 isSupportedForDocumentType[documentCreationType] = isCurrentPlaceholderSupportedForStrategy;
197 postfixNumbersSupportedForDocumentType[documentCreationType] = postfixNumbersSupportedStr;
198 }
199 }
200
201 DocumentPlaceholderViewConfig placeholderViewConfig = new DocumentPlaceholderViewConfig()
202 {
203 Key = placeholderKey,
204 Name = resourceStringName,
205 Info = resourceStringInfo,
206 Placeholders = string.Join(Environment.NewLine, placeholders.Value.Select(p => $"{_documentService.PlaceholderMarker}{p}{_documentService.PlaceholderMarker}")),
207 IsSupportedForDocumentType = isSupportedForDocumentType,
208 PostfixNumbersSupportedForDocumentType = postfixNumbersSupportedForDocumentType
209 };
210 PlaceholderViewConfigs.Add(placeholderViewConfig);
211 }
212 OnPropertyChanged(nameof(PlaceholderViewConfigs));
213 }

343 public void OnNavigatedTo(object parameter)
344 {
345 foreach(IDocumentStrategy strategy in _documentStrategies)
346 {
347 object[] items = strategy.GetItems();
348 bool isDataAvailable = items != null;
349 changeDocumentDataAvailableState(strategy.DocumentType, isDataAvailable);
350
351 bool isTemplateAvailable = !string.IsNullOrEmpty(strategy.TemplatePath) &&
352 File.Exists(strategy.TemplatePath) &&
353 Path.GetExtension(strategy.TemplatePath) == ".docx";
354 changeDocumentTemplateAvailableState(strategy.DocumentType, isTemplateAvailable);
355
356 if (!isDataAvailable || !isTemplateAvailable)
357 {
358 changeDocumentCreationRunningState(strategy.DocumentType, false);
359 changeDocumentCreationSuccessfulState(strategy.DocumentType, false);
360 }
361
362 // Find existing documents
363 string templateFileNamePostfix = _workspaceService?.Settings?.GetSettingValue<string>(WorkspaceSettings.GROUP_DOCUMENT_CREATION, WorkspaceSettings.SETTING_DOCUMENT_CREATION_TEMPLATE_FILENAME_POSTFIX) ?? string.Empty;
364 string documentOutputFolder = _workspaceService?.Settings?.GetSettingValue<string>(WorkspaceSettings.GROUP_DOCUMENT_CREATION, WorkspaceSettings.SETTING_DOCUMENT_CREATION_OUTPUT_FOLDER) ?? string.Empty;
365 documentOutputFolder = FilePathHelper.MakePathAbsolute(documentOutputFolder, _workspaceService?.PersistentPath);
366
367 string outputFileNameDocx = Path.GetFileNameWithoutExtension(strategy.TemplatePath);
368 outputFileNameDocx = outputFileNameDocx.Replace(templateFileNamePostfix, "") + ".docx";
369 string outputFilePathDocx = Path.Combine(documentOutputFolder, outputFileNameDocx);
370 string outputFilePathPdf = outputFilePathDocx.Replace(".docx", ".pdf", StringComparison.InvariantCultureIgnoreCase);
371 if(File.Exists(outputFilePathPdf))
372 {
373 changeLastDocumentFilePath(strategy.DocumentType, outputFilePathPdf);
374 changeDocumentCreationSuccessfulState(strategy.DocumentType, true); // necessary to show the path
375 }
376 else if(File.Exists(outputFilePathDocx))
377 {
378 changeLastDocumentFilePath(strategy.DocumentType, outputFilePathDocx);
379 changeDocumentCreationSuccessfulState(strategy.DocumentType, true); // necessary to show the path
380 }
381 }
