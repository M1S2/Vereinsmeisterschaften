212 private void initPlaceholderViewConfigs()
213 {
214 PlaceholderViewConfigs.Clear();
215 foreach(KeyValuePair<string, List<string>> placeholders in Placeholders.PlaceholderDict)
216 {
217 string placeholderKey = placeholders.Key;
218 ResourceManager rm = new ResourceManager(typeof(Properties.PlaceholderResources));
219 string resourceStringName = rm.GetString($"PlaceholderName{placeholderKey}") ?? "?";
220 string resourceStringInfo = rm.GetString($"PlaceholderInfo{placeholderKey}") ?? "?";
221
222 Dictionary<DocumentCreationTypes, bool> isSupportedForDocumentType = new Dictionary<DocumentCreationTypes, bool>();
223 Dictionary<DocumentCreationTypes, string> postfixNumbersSupportedForDocumentType = new Dictionary<DocumentCreationTypes, string>();
224 foreach (IDocumentStrategy strategy in _documentStrategies)
225 {
226 DocumentCreationTypes documentCreationType = strategy.DocumentType;
227 bool isCurrentPlaceholderSupportedForStrategy = strategy.SupportedPlaceholderKeys.Contains(placeholderKey);
228
229 int indexOfKey = strategy.SupportedPlaceholderKeys.IndexOf(placeholderKey);
230 int postfixNumbersSupported = (indexOfKey >= 0 && indexOfKey < strategy.PostfixNumbersSupported.Count) ? strategy.PostfixNumbersSupported[indexOfKey] : 0;
231 string postfixNumbersSupportedStr = postfixNumbersSupported > 0 ? postfixNumbersSupported.ToString() : string.Empty;
232
233 if (!isSupportedForDocumentType.ContainsKey(documentCreationType))
234 {
235 isSupportedForDocumentType.Add(documentCreationType, isCurrentPlaceholderSupportedForStrategy);
236 postfixNumbersSupportedForDocumentType.Add(documentCreationType, postfixNumbersSupportedStr);
237 }
238 else
239 {
240 isSupportedForDocumentType[documentCreationType] = isCurrentPlaceholderSupportedForStrategy;
241 postfixNumbersSupportedForDocumentType[documentCreationType] = postfixNumbersSupportedStr;
242 }
243 }
244
245 DocumentPlaceholderViewConfig placeholderViewConfig = new DocumentPlaceholderViewConfig()
246 {
247 Key = placeholderKey,
248 Name = resourceStringName,
249 Info = resourceStringInfo,
250 Placeholders = string.Join(Environment.NewLine, placeholders.Value.Select(p => $"{_documentService.PlaceholderMarker}{p}{_documentService.PlaceholderMarker}")),
251 IsSupportedForDocumentType = isSupportedForDocumentType,
252 PostfixNumbersSupportedForDocumentType = postfixNumbersSupportedForDocumentType
253 };
254 PlaceholderViewConfigs.Add(placeholderViewConfig);
255 }
256 OnPropertyChanged(nameof(PlaceholderViewConfigs));
257 }

308 {
309 int numCreatedPages = 0;

376 {
377 foreach(IDocumentStrategy strategy in _documentStrategies)
378 {
379 object[] items = strategy.GetItems();
380 bool isDataAvailable = items != null;
381 changeDocumentDataAvailableState(strategy.DocumentType, isDataAvailable);
382
383 bool isTemplateAvailable = !string.IsNullOrEmpty(strategy.TemplatePath) &&
384 System.IO.File.Exists(strategy.TemplatePath) &&
385 System.IO.Path.GetExtension(strategy.TemplatePath) == ".docx";
386 changeDocumentTemplateAvailableState(strategy.DocumentType, isTemplateAvailable);
387
388 if (!isDataAvailable || !isTemplateAvailable)
389 {
390 changeDocumentCreationRunningState(strategy.DocumentType, false);
391 changeDocumentCreationSuccessfulState(strategy.DocumentType, false);
392 }
393
394 // Find existing documents
395 string templateFileNamePostfix = _workspaceService?.Settings?.GetSettingValue<string>(WorkspaceSettings.GROUP_DOCUMENT_CREATION, WorkspaceSettings.SETTING_DOCUMENT_CREATION_TEMPLATE_FILENAME_POSTFIX) ?? string.Empty;
396 string documentOutputFolder = _workspaceService?.Settings?.GetSettingValue<string>(WorkspaceSettings.GROUP_DOCUMENT_CREATION, WorkspaceSettings.SETTING_DOCUMENT_CREATION_OUTPUT_FOLDER) ?? string.Empty;
397 documentOutputFolder = FilePathHelper.MakePathAbsolute(documentOutputFolder, _workspaceService?.PersistentPath);
398
399 string outputFileNameDocx = Path.GetFileNameWithoutExtension(strategy.TemplatePath);
400 outputFileNameDocx = outputFileNameDocx.Replace(templateFileNamePostfix, "") + ".docx";
401 string outputFilePathDocx = Path.Combine(documentOutputFolder, outputFileNameDocx);
402 string outputFilePathPdf = outputFilePathDocx.Replace(".docx", ".pdf", StringComparison.InvariantCultureIgnoreCase);
403 if(File.Exists(outputFilePathPdf))
404 {
405 changeLastDocumentFilePath(strategy.DocumentType, outputFilePathPdf);
406 changeDocumentCreationSuccessfulState(strategy.DocumentType, true); // necessary to show the path
407 }
408 else if(File.Exists(outputFilePathDocx))
409 {
410 changeLastDocumentFilePath(strategy.DocumentType, outputFilePathDocx);
411 changeDocumentCreationSuccessfulState(strategy.DocumentType, true); // necessary to show the path
412 }
