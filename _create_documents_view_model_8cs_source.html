193 private void initPlaceholderViewConfigs()
194 {
195 PlaceholderViewConfigs.Clear();
196 foreach(KeyValuePair<string, List<string>> placeholders in Placeholders.PlaceholderDict)
197 {
198 string placeholderKey = placeholders.Key;
199 ResourceManager rm = new ResourceManager(typeof(Properties.PlaceholderResources));
200 string resourceStringName = rm.GetString($"PlaceholderName{placeholderKey}") ?? "?";
201 string resourceStringInfo = rm.GetString($"PlaceholderInfo{placeholderKey}") ?? "?";
202
203 Dictionary<DocumentCreationTypes, bool> isSupportedForDocumentType = new Dictionary<DocumentCreationTypes, bool>();
204 Dictionary<DocumentCreationTypes, string> postfixNumbersSupportedForDocumentType = new Dictionary<DocumentCreationTypes, string>();
205 foreach (IDocumentStrategy strategy in _documentStrategies)
206 {
207 DocumentCreationTypes documentCreationType = strategy.DocumentType;
208 bool isCurrentPlaceholderSupportedForStrategy = strategy.SupportedPlaceholderKeys.Contains(placeholderKey);
209
210 int indexOfKey = strategy.SupportedPlaceholderKeys.IndexOf(placeholderKey);
211 int postfixNumbersSupported = (indexOfKey >= 0 && indexOfKey < strategy.PostfixNumbersSupported.Count) ? strategy.PostfixNumbersSupported[indexOfKey] : 0;
212 string postfixNumbersSupportedStr = postfixNumbersSupported > 0 ? postfixNumbersSupported.ToString() : string.Empty;
213
214 if (!isSupportedForDocumentType.ContainsKey(documentCreationType))
215 {
216 isSupportedForDocumentType.Add(documentCreationType, isCurrentPlaceholderSupportedForStrategy);
217 postfixNumbersSupportedForDocumentType.Add(documentCreationType, postfixNumbersSupportedStr);
218 }
219 else
220 {
221 isSupportedForDocumentType[documentCreationType] = isCurrentPlaceholderSupportedForStrategy;
222 postfixNumbersSupportedForDocumentType[documentCreationType] = postfixNumbersSupportedStr;
223 }
224 }
225
226 DocumentPlaceholderViewConfig placeholderViewConfig = new DocumentPlaceholderViewConfig()
227 {
228 Key = placeholderKey,
229 Name = resourceStringName,
230 Info = resourceStringInfo,
231 Placeholders = string.Join(Environment.NewLine, placeholders.Value.Select(p => $"{_documentService.PlaceholderMarker}{p}{_documentService.PlaceholderMarker}")),
232 IsSupportedForDocumentType = isSupportedForDocumentType,
233 PostfixNumbersSupportedForDocumentType = postfixNumbersSupportedForDocumentType
234 };
235 PlaceholderViewConfigs.Add(placeholderViewConfig);
236 }
237 OnPropertyChanged(nameof(PlaceholderViewConfigs));
238 }

279 private ICommand _createDocumentCommand;
283 public ICommand CreateDocumentCommand => _createDocumentCommand ?? (_createDocumentCommand = new RelayCommand<DocumentCreationTypes>(async (documentType) =>
284 {
285 changeDocumentCreationRunningState(documentType, true);
286 try
287 {
288 changeDocumentCreationSuccessfulState(documentType, false);
289 switch (documentType)
290 {
291 case DocumentCreationTypes.Certificates:
292 {
293 object filterParam = null;
294 switch (PersonStartFilter)
295 {
296 case PersonStartFilters.None: filterParam = null; break;
297 case PersonStartFilters.Person: filterParam = FilteredPerson; break;
298 case PersonStartFilters.SwimmingStyle: filterParam = FilteredSwimmingStyle; break;
299 case PersonStartFilters.CompetitionID: filterParam = FilteredCompetitionID; break;
300 default: break;
301 }
302
303 _documentService.SetCertificateCreationFilters(PersonStartFilter, filterParam);
304 NumberCreatedCertificates = await _documentService.CreateDocument(DocumentCreationTypes.Certificates);
305 break;
306 }
307 case DocumentCreationTypes.OverviewList:
308 {
309 await _documentService.CreateDocument(DocumentCreationTypes.OverviewList);
310 break;
311 }
312 case DocumentCreationTypes.RaceStartList:
313 {
314 await _documentService.CreateDocument(DocumentCreationTypes.RaceStartList);
315 break;
316 }
317 case DocumentCreationTypes.TimeForms:
318 {
319 NumberCreatedTimeForms = await _documentService.CreateDocument(DocumentCreationTypes.TimeForms);
320 break;
321 }
322 case DocumentCreationTypes.ResultList:
323 {
324 await _documentService.CreateDocument(DocumentCreationTypes.ResultList);
325 break;
326 }
327 case DocumentCreationTypes.ResultListDetail:
328 {
329 await _documentService.CreateDocument(DocumentCreationTypes.ResultListDetail);
330 break;
331 }
332 default: throw new ArgumentOutOfRangeException(nameof(documentType), documentType, "Unknown document type for creation command.");
333 }
334 changeDocumentCreationSuccessfulState(documentType, true);
335 }
336 catch (Exception ex)

338 await _dialogCoordinator.ShowMessageAsync(this, Resources.ErrorString, ex.Message);

345 public void OnNavigatedTo(object parameter)
346 {
347 foreach(IDocumentStrategy strategy in _documentStrategies)
348 {
349 object[] items = strategy.GetItems();
350 bool isDataAvailable = items != null;
351 changeDocumentDataAvailableState(strategy.DocumentType, isDataAvailable);
352
353 bool isTemplateAvailable = !string.IsNullOrEmpty(strategy.TemplatePath) &&
354 System.IO.File.Exists(strategy.TemplatePath) &&
355 System.IO.Path.GetExtension(strategy.TemplatePath) == ".docx";
356 changeDocumentTemplateAvailableState(strategy.DocumentType, isTemplateAvailable);
357
358 if (!isDataAvailable || !isTemplateAvailable)
359 {
360 changeDocumentCreationRunningState(strategy.DocumentType, false);
361 changeDocumentCreationSuccessfulState(strategy.DocumentType, false);
362 }
363 }
364
365 initPlaceholderViewConfigs();
