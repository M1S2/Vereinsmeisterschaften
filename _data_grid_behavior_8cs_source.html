180 DataGrid grid = (DataGrid)sender;
181 string path = e.Column.SortMemberPath;
182 if (string.IsNullOrEmpty(path)) { return; }
183
184 bool shiftPressed = Keyboard.Modifiers.HasFlag(ModifierKeys.Shift);
185 e.Handled = true;
186
187 // Get the new sort direction based on the current sort direction
188 ListSortDirection? newDirection = e.Column.SortDirection switch
189 {
190 null => ListSortDirection.Ascending,
191 ListSortDirection.Ascending => ListSortDirection.Descending,
192 ListSortDirection.Descending => null,
193 _ => null
194 };
195
196 // If Shift is NOT pressed → clear all previous sortings
197 if (!shiftPressed)
198 {
199 foreach (DataGridColumn c in grid.Columns)
200 {
201 if (!ReferenceEquals(c, e.Column))
202 {
203 c.SortDirection = null;
204 }
205 }
206 grid.Items.SortDescriptions.Clear();
207 }
208
209 // Remove the column from SortDescriptions if it exists
210 SortDescription existingSortDescription = grid.Items.SortDescriptions.FirstOrDefault(sd => sd.PropertyName == path);
211 if (!string.IsNullOrEmpty(existingSortDescription.PropertyName))
212 {
213 grid.Items.SortDescriptions.Remove(existingSortDescription);
214 }
215
216 if (newDirection != null)
217 {
218 // Add new SortDescription with the new direction
219 grid.Items.SortDescriptions.Add(new SortDescription(path, newDirection.Value));
220 e.Column.SortDirection = newDirection;
221 }
222 else
223 {
224 // No sorting → reset column
225 e.Column.SortDirection = null;
226 }

228 // Apply sorting
