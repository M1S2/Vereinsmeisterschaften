117 case ItemOrderingsCertificate.ByFirstNameAscending: starts = starts.OrderBy(s => s.PersonObj?.FirstName).ToList(); break;
118 case ItemOrderingsCertificate.ByFirstNameDescending: starts = starts.OrderByDescending(s => s.PersonObj?.FirstName).ToList(); break;
119 case ItemOrderingsCertificate.ByCompetitionAscending: starts = starts.OrderBy(s => s.CompetitionObj?.Id).ToList(); break;
120 case ItemOrderingsCertificate.ByCompetitionDescending: starts = starts.OrderByDescending(s => s.CompetitionObj?.Id).ToList(); break;
