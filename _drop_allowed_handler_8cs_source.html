48 private bool dropAllowed(IDropInfo dropInfo)
49 {
50 ICollection sourceCollection = dropInfo.DragInfo.SourceCollection as ICollection;
51 ICollection targetCollection = dropInfo.TargetCollection as ICollection;
52
53 PersonStart dragItem = dropInfo.DragInfo.SourceItem as PersonStart;
54 PersonStart dropItem = dropInfo.TargetItem as PersonStart;
55
56 bool dropAllowed = (targetCollection == sourceCollection ||
57 targetCollection != sourceCollection && targetCollection.Count + 1 <= MaxItemsInTargetCollection);
58 if (!CanAcceptData(dropInfo))
59 {
60 return false;
61 }
62 else if (dragItem != null && dropItem != null)
63 {
64 return dropAllowed &&
65 dragItem.Style == dropItem.Style &&
66 dragItem.CompetitionObj?.Distance == dropItem.CompetitionObj?.Distance;
67 }
68 else if (dragItem != null && dropItem == null && targetCollection.Count == 0)
69 {
70 return dropAllowed;
71 }
72 else if (dragItem != null && dropItem == null && targetCollection.Count > 0)
73 {
74 PersonStart firstStart = (targetCollection as IList)?.Cast<PersonStart>().FirstOrDefault();
75
76 return dropAllowed &&
77 dragItem.Style == firstStart?.Style &&
78 dragItem.CompetitionObj?.Distance == firstStart?.CompetitionObj?.Distance;
