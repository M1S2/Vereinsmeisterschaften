222 public void CreateFriendGroupViewModel()
223 {
224 _pauseFriendsCollectionChangedEvent = true;
225 FriendGroups.Clear();
226 _pauseFriendsCollectionChangedEvent = false;
227
228 // Collect all unique group IDs from all persons
229 List<int> allGroupIds = People.SelectMany(p => p.FriendGroupIDs ?? Enumerable.Empty<int>()).Distinct().OrderBy(id => id).ToList();
230
231 foreach (int groupId in allGroupIds)
232 {
233 // Create view model entry for the group
234 FriendGroupViewModel group = new FriendGroupViewModel()
235 {
236 GroupId = groupId,
237 AvailableFriends = People,
238 // Get all persons belonging to this group
239 Friends = new ObservableCollection<Person>(People.Where(p => p.FriendGroupIDs.Contains(groupId)))
240 };
241 group.Friends.CollectionChanged += (s, e) => Friends_CollectionChanged(s, e, group);
242 FriendGroups.Add(group);
243 }
244 }

246 private bool _pauseFriendsCollectionChangedEvent = false;
247 private void Friends_CollectionChanged(object sender, System.Collections.Specialized.NotifyCollectionChangedEventArgs e, FriendGroupViewModel groupViewModel)
248 {
249 if (_pauseFriendsCollectionChangedEvent) { return; }
250
251 int groupId = groupViewModel.GroupId;
252 switch (e.Action)
253 {
254 case System.Collections.Specialized.NotifyCollectionChangedAction.Add:
255 {
256 foreach (Person newPerson in e.NewItems)
257 {
258 if (!newPerson.FriendGroupIDs.Contains(groupId))
259 {
260 newPerson.FriendGroupIDs.Add(groupId);
261 }
262 }
263 break;
264 }
265 case System.Collections.Specialized.NotifyCollectionChangedAction.Remove:
266 {
267 foreach (Person oldPerson in e.OldItems)
268 {
269 if (oldPerson.FriendGroupIDs.Contains(groupId))
270 {
271 oldPerson.FriendGroupIDs.Remove(groupId);
272 }
273 }
274 break;
275 }
276 default: break;
277 }
278 _personService.UpdateAllFriendReferencesFromFriendGroupIDs();
279 }
