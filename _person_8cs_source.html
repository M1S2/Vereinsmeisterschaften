586 private bool CompareDictionaries<TKey, TValue>(Dictionary<TKey, TValue> dict1, Dictionary<TKey, TValue> dict2)
587 {
588 if (dict1 == dict2) return true;
589 if ((dict1 == null) || (dict2 == null)) return false;
590 if (dict1.Count != dict2.Count) return false;
591
592 EqualityComparer<TValue> valueComparer = EqualityComparer<TValue>.Default;
593
594 foreach (KeyValuePair<TKey, TValue> kvp in dict1)
595 {
596 TValue value2;
597 if (!dict2.TryGetValue(kvp.Key, out value2)) return false;
598 if (!valueComparer.Equals(kvp.Value, value2)) return false;
599 }
600 return true;
601 }
