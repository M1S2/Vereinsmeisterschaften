431 private bool CompareDictionaries<TKey, TValue>(Dictionary<TKey, TValue> dict1, Dictionary<TKey, TValue> dict2)
432 {
433 if (dict1 == dict2) return true;
434 if ((dict1 == null) || (dict2 == null)) return false;
435 if (dict1.Count != dict2.Count) return false;
436
437 EqualityComparer<TValue> valueComparer = EqualityComparer<TValue>.Default;
438
439 foreach (KeyValuePair<TKey, TValue> kvp in dict1)
440 {
441 TValue value2;
442 if (!dict2.TryGetValue(kvp.Key, out value2)) return false;
443 if (!valueComparer.Equals(kvp.Value, value2)) return false;
444 }
445 return true;
446 }
