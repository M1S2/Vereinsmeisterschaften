101 public async Task<bool> Load(string path, CancellationToken cancellationToken)
102 {
103 bool importingResult = false;
104 Exception exception = null;
105 await Task.Run(() =>
106 {
107 try
108 {
109 if (!File.Exists(path))
110 {
111 OnFileProgress?.Invoke(this, 0);
112 ClearAll();
113 OnFileProgress?.Invoke(this, 100);
114 }
115 else
116 {
117 List<Person> list = _fileService.LoadFromCsv<Person>(path, cancellationToken, Person.SetPropertyFromString, OnFileProgress, (header) =>
118 {
119 return PropertyNameLocalizedStringHelper.FindProperty(typeof(Person), header);
120 });
121 _personList = new ObservableCollection<Person>();
122 foreach (Person person in list)
123 {
124 AddPerson(person);
125 }
126 }
127
128 _personListOnLoad = _personList.ToList().ConvertAll(p => new Person(p));
129
130 PersistentPath = path;
131 importingResult = true;
132 }
133 catch (OperationCanceledException)
134 {
135 importingResult = false;
136 }
137 catch (Exception ex)
138 {
139 exception = ex;
140 }
141 });
142 OnFileFinished?.Invoke(this, null);
143 if (exception != null) { throw exception; }
144 return importingResult;
145 }

155 public async Task<bool> Save(CancellationToken cancellationToken, string path = "")
156 {
157 if(string.IsNullOrEmpty(path)) { path = PersistentPath; }
158
159 bool saveResult = false;
160 Exception exception = null;
161 await Task.Run(() =>
162 {
163 try
164 {
165 _fileService.SaveToCsv(path, _personList.ToList(), cancellationToken, OnFileProgress, (data, parentObject, currentProperty) =>
166 {
167 if (data is bool dataBool)
168 {
169 // Get the corresponding IsActive flag for the swimming style and set the file content accordingly
170 bool isActive = true;
171 switch(currentProperty.Name)
172 {
173 case nameof(Person.Breaststroke): isActive = (parentObject as Person)?.Starts[SwimmingStyles.Breaststroke]?.IsActive ?? true; break;
174 case nameof(Person.Freestyle): isActive = (parentObject as Person)?.Starts[SwimmingStyles.Freestyle]?.IsActive ?? true; break;
175 case nameof(Person.Backstroke): isActive = (parentObject as Person)?.Starts[SwimmingStyles.Backstroke]?.IsActive ?? true; break;
176 case nameof(Person.Butterfly): isActive = (parentObject as Person)?.Starts[SwimmingStyles.Butterfly]?.IsActive ?? true; break;
177 case nameof(Person.Medley): isActive = (parentObject as Person)?.Starts[SwimmingStyles.Medley]?.IsActive ?? true; break;
178 case nameof(Person.WaterFlea): isActive = (parentObject as Person)?.Starts[SwimmingStyles.WaterFlea]?.IsActive ?? true; break;
179 }
180 return dataBool ? (isActive ? "X" : Person.START_INACTIVE_MARKER_STRING) : "";
181 }
182 else if (data is Enum dataEnum)
183 {
184 return EnumCoreLocalizedStringHelper.Convert(dataEnum);
185 }
186 else
187 {
188 return data.ToString();
189 }
190 },
191 (header, type) =>
192 {
193 return PropertyNameLocalizedStringHelper.Convert(typeof(Person), header);
194 });
195
196 _personListOnLoad = _personList.ToList().ConvertAll(p => new Person(p));
197 saveResult = true;
198 }
199 catch (OperationCanceledException)
200 {
201 saveResult = false;
202 }
203 catch (Exception ex)
204 {
205 exception = ex;
206 }
207 });
208 OnFileFinished?.Invoke(this, null);
209 if (exception != null) { throw exception; }
210 return saveResult;
211 }

289 private bool _isupdatingScores = false; // Flag to avoid recursive calls of the Person_PropertyChanged event. During UpdateScoresForPerson the Score of the PersonStart is changed which would raise a property changed event of the Starts property of the Person again.
298 private void Person_PropertyChanged(object sender, PropertyChangedEventArgs e)
299 {
300 if (e.PropertyName == nameof(Person.Starts) && sender is Person senderPerson && _scoreService != null && !_isupdatingScores)
301 {
302 _isupdatingScores = true;
303 _scoreService.UpdateScoresForPerson(senderPerson);
304 _isupdatingScores = false;
305 }
306
307 switch(e.PropertyName)
308 {
309 case nameof(Person.Name):
310 case nameof(Person.FirstName):
311 // Duplicates depends on the basic parameters of the person (First name, Name, Birth year, Gender)
312 UpdateHasDuplicatesForPersons();
313 break;
314 case nameof(Person.Gender):
315 case nameof(Person.BirthYear):
316 // Duplicates depends on the basic parameters of the person (First name, Name, Birth year, Gender)
317 UpdateHasDuplicatesForPersons();
318 // Update the competitions for the person. They depend on the gender and birth year.
319 _competitionService.UpdateAllCompetitionsForPerson(sender as Person);
320 break;
321 case nameof(Person.Starts):
322 OnPropertyChanged(nameof(PersonStarts));
323 break;
324 default: break;
325 }
326
327 OnPropertyChanged(nameof(HasUnsavedChanges));

343 public void UpdateHasDuplicatesForPersons()
344 {
345 PersonBasicEqualityComparer basicPersonComparer = new PersonBasicEqualityComparer();
346 IEnumerable<IGrouping<Person, Person>> duplicateGroups = _personList.GroupBy(p => p, basicPersonComparer).Where(g => g.Count() > 1);
347
348 // Reset all HasDuplicates flags
349 foreach (Person p in _personList) { p.HasDuplicates = false; }
350
351 // Set HasDuplicates flag for all persons that are in a duplicate group
352 foreach (IGrouping<Person, Person> duplicateGroup in duplicateGroups)
353 {
354 foreach (Person person in duplicateGroup) { person.HasDuplicates = true; }
355 }

372 public List<PersonStart> GetAllPersonStarts(PersonStartFilters filter = PersonStartFilters.None, object filterParameter = null)
373 {
374 List<PersonStart> allPersonStarts = new List<PersonStart>();
375 foreach (Person person in _personList)
376 {
377 allPersonStarts.AddRange(person.Starts.Values.Cast<PersonStart>().Where(s => s != null));
378 }
379
380 switch (filter)
381 {
382 case PersonStartFilters.None:
383 return allPersonStarts;
384 case PersonStartFilters.Person:
385 PersonBasicEqualityComparer personBasicComparer = new PersonBasicEqualityComparer();
386 return allPersonStarts.Where(s => personBasicComparer.Equals(s.PersonObj, (Person)filterParameter)).ToList();
387 case PersonStartFilters.SwimmingStyle:
388 return allPersonStarts.Where(s => s.Style == (SwimmingStyles)filterParameter).ToList();
389 case PersonStartFilters.CompetitionID:
390 return allPersonStarts.Where(s => s.CompetitionObj != null && s.CompetitionObj.Id == (int)filterParameter).ToList();
391 default: return allPersonStarts;
392 }
393 }
