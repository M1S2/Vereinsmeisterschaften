246 public List<Person> CheckForDuplicatePerson()

279 public List<PersonStart> GetAllPersonStarts(PersonStartFilters filter = PersonStartFilters.None, object filterParameter = null)
280 {
281 List<PersonStart> allPersonStarts = new List<PersonStart>();
282 foreach (Person person in _personList)
283 {
284 allPersonStarts.AddRange(person.Starts.Values.Cast<PersonStart>().Where(s => s != null));
285 }
286
287 switch (filter)
288 {
289 case PersonStartFilters.None: return allPersonStarts;
290 case PersonStartFilters.Person: return allPersonStarts.Where(s => s.PersonObj == (Person)filterParameter).ToList();
291 case PersonStartFilters.SwimmingStyle: return allPersonStarts.Where(s => s.Style == (SwimmingStyles)filterParameter).ToList();
292 case PersonStartFilters.CompetitionID: return allPersonStarts.Where(s => s.CompetitionObj != null && s.CompetitionObj.ID == (int)filterParameter).ToList();
293 default: return allPersonStarts;
294 }
295 }
