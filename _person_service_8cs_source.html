78 public async Task<bool> Load(string path, CancellationToken cancellationToken)
79 {
80 bool importingResult = false;
81 Exception exception = null;
82 await Task.Run(() =>
83 {
84 try
85 {
86 if (!File.Exists(path))
87 {
88 OnFileProgress?.Invoke(this, 0);
89 ClearAll();
90 OnFileProgress?.Invoke(this, 100);
91 }
92 else
93 {
94 List<Person> list = _fileService.LoadFromCsv<Person>(path, cancellationToken, Person.SetPropertyFromString, OnFileProgress);
95 _personList = new ObservableCollection<Person>();
96 foreach (Person person in list)
97 {
98 AddPerson(person);
99 }
100 }
101
102 _personListOnLoad = _personList.ToList().ConvertAll(p => new Person(p));
103
104 PersistentPath = path;
105 importingResult = true;
106 }
107 catch (OperationCanceledException)
108 {
109 importingResult = false;
110 }
111 catch (Exception ex)
112 {
113 exception = ex;
114 }
115 });
116 OnFileFinished?.Invoke(this, null);
117 if (exception != null) { throw exception; }
118 return importingResult;
119 }

264 public List<Person> CheckForDuplicatePerson()

297 public List<PersonStart> GetAllPersonStarts(PersonStartFilters filter = PersonStartFilters.None, object filterParameter = null)
298 {
299 List<PersonStart> allPersonStarts = new List<PersonStart>();
300 foreach (Person person in _personList)
301 {
302 allPersonStarts.AddRange(person.Starts.Values.Cast<PersonStart>().Where(s => s != null));
303 }
304
305 switch (filter)
306 {
307 case PersonStartFilters.None: return allPersonStarts;
308 case PersonStartFilters.Person: return allPersonStarts.Where(s => s.PersonObj == (Person)filterParameter).ToList();
309 case PersonStartFilters.SwimmingStyle: return allPersonStarts.Where(s => s.Style == (SwimmingStyles)filterParameter).ToList();
310 case PersonStartFilters.CompetitionID: return allPersonStarts.Where(s => s.CompetitionObj != null && s.CompetitionObj.ID == (int)filterParameter).ToList();
311 default: return allPersonStarts;
312 }
313 }
