90 public async Task<bool> Load(string path, CancellationToken cancellationToken)
91 {
92 bool importingResult = false;
93 Exception exception = null;
94 await Task.Run(() =>
95 {
96 try
97 {
98 if (!File.Exists(path))
99 {
100 OnFileProgress?.Invoke(this, 0);
101 ClearAll();
102 OnFileProgress?.Invoke(this, 100);
103 }
104 else
105 {
106 List<Person> list = _fileService.LoadFromCsv<Person>(path, cancellationToken, Person.SetPropertyFromString, OnFileProgress, (header) =>
107 {
108 return PropertyNameLocalizedStringHelper.FindProperty(typeof(Person), header);
109 });
110 _personList = new ObservableCollection<Person>();
111 foreach (Person person in list)
112 {
113 AddPerson(person);
114 }
115 }
116
117 _personListOnLoad = _personList.ToList().ConvertAll(p => new Person(p));
118
119 PersistentPath = path;
120 importingResult = true;
121 }
122 catch (OperationCanceledException)
123 {
124 importingResult = false;
125 }
126 catch (Exception ex)
127 {
128 exception = ex;
129 }
130 });
131 OnFileFinished?.Invoke(this, null);
132 if (exception != null) { throw exception; }
133 return importingResult;
134 }

305 public List<Person> CheckForDuplicatePerson()
306 {
307 List<Person> tmpPersonList = new List<Person>();
308 List<Person> duplicates = new List<Person>();
309 PersonBasicEqualityComparer basicPersonComparer = new PersonBasicEqualityComparer();
310 foreach (Person person in _personList)

316 else
317 {
318 duplicates.Add(person);
319 }
320 }
321 return duplicates.Distinct().ToList();

338 public List<PersonStart> GetAllPersonStarts(PersonStartFilters filter = PersonStartFilters.None, object filterParameter = null)
339 {
340 List<PersonStart> allPersonStarts = new List<PersonStart>();
341 foreach (Person person in _personList)
342 {
343 allPersonStarts.AddRange(person.Starts.Values.Cast<PersonStart>().Where(s => s != null));
344 }
345
346 switch (filter)
347 {
348 case PersonStartFilters.None: return allPersonStarts;
349 case PersonStartFilters.Person: return allPersonStarts.Where(s => s.PersonObj == (Person)filterParameter).ToList();
350 case PersonStartFilters.SwimmingStyle: return allPersonStarts.Where(s => s.Style == (SwimmingStyles)filterParameter).ToList();
351 case PersonStartFilters.CompetitionID: return allPersonStarts.Where(s => s.CompetitionObj != null && s.CompetitionObj.Id == (int)filterParameter).ToList();
352 default: return allPersonStarts;
353 }
354 }
