227 private void recalculateHighlightedPersonStarts()
228 {
229 List<PersonStart> personStarts = _personService.GetAllPersonStarts();
230 foreach (PersonStart personStart in personStarts)
231 {
232 switch (HighlightPersonStartMode)
233 {
234 case HighlightPersonStartModes.Person: personStart.IsHighlighted = personStart.PersonObj.Equals(HighlightedPerson); break;
235 case HighlightPersonStartModes.SwimmingStyle: personStart.IsHighlighted = personStart.Style.Equals(HighlightedSwimmingStyle); break;
236 case HighlightPersonStartModes.Gender: personStart.IsHighlighted = personStart.PersonObj.Gender.Equals(HighlightedGender); break;
237 case HighlightPersonStartModes.Distance: personStart.IsHighlighted = personStart.CompetitionObj?.Distance == HighlightedDistance; break;
238 case HighlightPersonStartModes.None: personStart.IsHighlighted = false; break;
239 default: personStart.IsHighlighted = false; break;
240 }

314 private ICommand _calculateRacesVariantsCommand;
318 public ICommand CalculateRacesVariantsCommand => _calculateRacesVariantsCommand ?? (_calculateRacesVariantsCommand = new RelayCommand(async() =>
319 {
320 double nextProgressLevel = 0.0;
321 ProgressDelegate onProgress = (sender, progress, currentStep) =>
322 {
323 if (progress >= nextProgressLevel)
324 {
325 nextProgressLevel += 0.5; // Only report all 0.5%. This is enough.
326 _progressController?.SetProgress(progress / 100);
327 _progressController?.SetMessage(string.IsNullOrEmpty(currentStep) ? $"{progress:F1}%" : $"{currentStep}: {progress:F1}%");
328 }
329 };
330
331 CancellationTokenSource cancellationTokenSource = new CancellationTokenSource();
332 // Open a user dialog to show the current calculation progress
333 _progressController = await _dialogCoordinator.ShowProgressAsync(this, Properties.Resources.CalculateRacesVariantsString, "", true);
334 _progressController.Canceled += (sender, e) => cancellationTokenSource.Cancel();
335
336 try
337 {
338 await _raceService.CalculateRacesVariants(cancellationTokenSource.Token, onProgress);
339 recalculateVariantIDs();
340
341 await _progressController?.CloseAsync();
342
343 int numberVariants = AllRacesVariants.Count;
344 ushort numberRequestedVariants = _workspaceService?.Settings?.GetSettingValue<ushort>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_NUM_RACE_VARIANTS_AFTER_CALCULATION) ?? 0;
345 if (numberVariants < numberRequestedVariants)
346 {
347 // Less variants than required are calculated
348 await _dialogCoordinator.ShowMessageAsync(this, Properties.Resources.CalculateRacesVariantsString, string.Format(Properties.Resources.CalculationWarningTooLessVariantsString, numberVariants, numberRequestedVariants));
349 }
350 }
351 catch (OperationCanceledException)
352 {
353 await _progressController?.CloseAsync();
354 }
355 catch (Exception ex)
356 {
357 await _progressController?.CloseAsync();
358 await _dialogCoordinator.ShowMessageAsync(this, Properties.Resources.ErrorString, ex.Message);
359 }
360 }));
361
362 #endregion
