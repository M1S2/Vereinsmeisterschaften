90 public async Task<ObservableCollection<RacesVariant>> CalculateRacesVariants(CancellationToken cancellationToken, ProgressDelegate onProgress = null)
91 {
92 // Collect all starts
93 _competitionService.UpdateAllCompetitionsForPerson();
94 List<PersonStart> starts = _personService.GetAllPersonStarts();
95
96 // Create groups of competitions with same style and distance
97 Dictionary<(SwimmingStyles, ushort), List<PersonStart>> groupedValuesStarts = new Dictionary<(SwimmingStyles, ushort), List<PersonStart>>();
98 for (int i = 0; i < starts.Count; i++)
99 {
100 if (!starts[i].IsCompetitionObjAssigned) { continue; }
101
102 (SwimmingStyles, ushort) key = (starts[i].CompetitionObj.SwimmingStyle, starts[i].CompetitionObj.Distance);
103 if (!groupedValuesStarts.ContainsKey(key))
104 {
105 groupedValuesStarts.Add(key, new List<PersonStart>());
106 }
107 else if (groupedValuesStarts[key] == null)
108 {
109 groupedValuesStarts[key] = new List<PersonStart>();
110 }
111 groupedValuesStarts[key].Add(starts[i]);
112 }
113
114 ushort numSwimLanes = _workspaceService?.Settings?.GetSettingValue<ushort>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_NUMBER_OF_SWIM_LANES) ?? 0;
115 ushort numberRequestedVariants = _workspaceService?.Settings?.GetSettingValue<ushort>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_NUM_RACE_VARIANTS_AFTER_CALCULATION) ?? 0;
116 int maxRaceVariantCalculationLoops = _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_MAX_CALCULATION_LOOPS) ?? 0;
117 double minRacesVariantsScore = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_MIN_RACES_VARIANTS_SCORE) ?? 0;
118 int numberOfResultsToGenerate = Math.Max(0, numberRequestedVariants - AllRacesVariants.Count(r => r.KeepWhileRacesCalculation));
119
120 RacesVariantsGenerator generator = new RacesVariantsGenerator(new Progress<double>(progress => onProgress?.Invoke(this, (float)progress, "")),
121 numberOfResultsToGenerate,
122 maxRaceVariantCalculationLoops,
123 minRacesVariantsScore,
124 numSwimLanes);
125 List<RacesVariant> tmpRacesVariants = await generator.GenerateBestRacesAsync(groupedValuesStarts.Values.ToList(), cancellationToken);
126
127 if (!cancellationToken.IsCancellationRequested)
128 {
129 List<RacesVariant> racesToDelete = AllRacesVariants.Where(r => !r.KeepWhileRacesCalculation).ToList();
130 racesToDelete.ForEach(r => AllRacesVariants.Remove(r));
131 tmpRacesVariants.ForEach(AddRacesVariant);
132 SortVariantsByScore();
133 OnPropertyChanged(nameof(PersistedRacesVariant));
134 }
135 return AllRacesVariants;
136 }

148 public async Task<bool> Load(string path, CancellationToken cancellationToken)
149 {
150 bool importingResult = false;
151 Exception exception = null;
152
153 SynchronizationContext uiContext = SynchronizationContext.Current;
154 await Task.Run(() =>
155 {
156 try
157 {
158 if (!File.Exists(path))
159 {
160 OnFileProgress?.Invoke(this, 0);
161 uiContext.Send((d) => { ClearAllRacesVariants(); }, null);
162 OnFileProgress?.Invoke(this, 100);
163 }
164 else
165 {
166 List<Race> raceList = _fileService.LoadFromCsv<Race>(path, cancellationToken, setRacePropertyFromString, OnFileProgress, (header) =>
167 {
168 return PropertyNameLocalizedStringHelper.FindProperty(typeof(Race), header);
169 });
170 RacesVariant bestRacesVariant = new RacesVariant(raceList, _workspaceService);
171 bestRacesVariant.IsPersistent = true;
172 uiContext.Send((d) =>
173 {
174 ClearAllRacesVariants();
175 AddRacesVariant(bestRacesVariant);
176 }, null);
177 }
178
179 if (PersistedRacesVariant == null) { _persistedRacesVariantOnLoad = null; }
180 else { _persistedRacesVariantOnLoad = new RacesVariant(PersistedRacesVariant, true, true, _workspaceService); }
181
182 PersistentPath = path;
183 importingResult = true;
184 }
185 catch (OperationCanceledException)
186 {
187 importingResult = false;
188 }
189 catch (Exception ex)
190 {
191 exception = ex;
192 }
193 });
194 OnFileFinished?.Invoke(this, null);
195 if (exception != null) { throw exception; }
196 return importingResult;
197 }

199 private int _lastParsedDistance = 0;
200 private SwimmingStyles _lastParsedStyle = SwimmingStyles.Unknown;
207 private void setRacePropertyFromString(Race dataObj, string propertyName, string value)
208 {
209 if (string.IsNullOrEmpty(value)) return;
210
211 switch (propertyName)
212 {
213 case nameof(Race.Distance): _lastParsedDistance = int.Parse(value); break;
214 case nameof(Race.Style): if (EnumCoreLocalizedStringHelper.TryParse(value, out SwimmingStyles parsedStyle)) { _lastParsedStyle = parsedStyle; } break;
215 default:
216 {
217 if (dataObj.Starts == null) { dataObj.Starts = new ObservableCollection<PersonStart>(); }
218 if(_lastParsedDistance != 0 && _lastParsedStyle != SwimmingStyles.Unknown)
219 {
220 string[] nameParts = value.Split(',');
221 string firstName = nameParts.Length > 0 ? nameParts[0].Trim() : "";
222 string name = nameParts.Length > 1 ? nameParts[1].Trim() : "";
223 PersonStart matchingStart = _personService.GetAllPersonStarts().Where(s => s.Style == _lastParsedStyle &&
224 s.CompetitionObj?.Distance == _lastParsedDistance &&
225 s.PersonObj?.FirstName == firstName &&
226 s.PersonObj?.Name == name).FirstOrDefault();
227 if (matchingStart != null)
228 {
229 dataObj.Starts.Add(matchingStart);
230 }
231 }
232 break;
233 }
234 }

245 public async Task<bool> Save(CancellationToken cancellationToken, string path = "")
246 {
247 if (string.IsNullOrEmpty(path)) { path = PersistentPath; }
248
249 bool saveResult = false;
250 Exception exception = null;
251 await Task.Run(() =>
252 {
253 try
254 {
255 if (PersistedRacesVariant == null)
256 {
257 if (File.Exists(path)) { File.Delete(path); }
258 _persistedRacesVariantOnLoad = null;
259 }
260 else
261 {
262 int maxNumberStarts = PersistedRacesVariant.Races.Count == 0 ? 0 : PersistedRacesVariant.Races.Select(r => r.Starts.Count).Max();
263 _fileService.SaveToCsv(path, PersistedRacesVariant.Races.ToList(), cancellationToken, OnFileProgress,
264 (data, parentObject, currentProperty) =>
265 {
266 if (data is IList<PersonStart> dataList)
267 {
268 return string.Join(";", dataList.Select(p => p.PersonObj?.FirstName + ", " + p.PersonObj?.Name));
269 }
270 else if (data is Enum dataEnum)
271 {
272 return EnumCoreLocalizedStringHelper.Convert(dataEnum);
273 }
274 else
275 {
276 return data.ToString();
277 }
278 },
279 (header, headerType) =>
280 {
281 if (headerType == typeof(ObservableCollection<PersonStart>))
282 {
283 string formatedHeader = "";
284 for(int i = 1; i <= maxNumberStarts; i++)
285 {
286 formatedHeader += $"{PropertyNameLocalizedStringHelper.Convert(typeof(Race), "Person")} {i};";
287 }
288 formatedHeader = formatedHeader.Trim(';');
289 return formatedHeader;
290 }
291 else
292 {
293 return PropertyNameLocalizedStringHelper.Convert(typeof(Race), header);
294 }
295 });
296
297 _persistedRacesVariantOnLoad = new RacesVariant(PersistedRacesVariant, true, true, _workspaceService);
298 }
299 saveResult = true;
300 }
301 catch (OperationCanceledException)
302 {
303 saveResult = false;
304 }
305 catch (Exception ex)
306 {
307 exception = ex;
308 }
309 });
310 OnFileFinished?.Invoke(this, null);
311 if (exception != null) { throw exception; }
312 return saveResult;
313 }

452 public void CleanupRacesVariants(bool removeInactiveStarts = true)
453 {
454 List<PersonStart> validPersonStarts = _personService.GetAllPersonStarts();
455
456 foreach (RacesVariant racesVariant in AllRacesVariants)
457 {
458 foreach(Race race in racesVariant.Races)
459 {
460 // Find all starts in this race that are no longer part of the valid PersonStarts and remove them from the Race
461 List<PersonStart> startsToDelete = race.Starts.Except(validPersonStarts, new PersonStartWithoutIsActiveEqualityComparer()).ToList();
462 if (removeInactiveStarts)
463 {
464 // Also remove all inactive starts
465 startsToDelete.AddRange(race.Starts.Where(s => !s.IsActive));
466 // Make sure to remove duplicate elements
467 startsToDelete = startsToDelete.Distinct(new PersonStartWithoutIsActiveEqualityComparer()).ToList();
468 }
469 // Remove the starts from the race
470 startsToDelete.ForEach(s => race.Starts.Remove(s));
471 }
472
473 // Find all empty races and remove them from the variant
474 List<Race> racesToDelete = racesVariant.Races.Where(r => r.Starts.Count == 0).ToList();
475 racesToDelete.ForEach(r => racesVariant.Races.Remove(r));
476 }
477 }
