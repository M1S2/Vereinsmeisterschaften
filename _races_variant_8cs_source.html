267 public double Score
268 {
269 get
270 {
271 double weightSingleStarts = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SINGLE_STARTS) ?? 5;
272 double weightSameStyleSequence = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SAME_STYLE_SEQUENCE) ?? 5;
273 double weightPersonStartPauses = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_PERSON_START_PAUSES) ?? 65;
274 double weightStyleOrder = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_STYLE_ORDER) ?? 10;
275 double weightStartGenders = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_START_GENDERS) ?? 5;
276 double sumWeights = weightSingleStarts + weightSameStyleSequence + weightPersonStartPauses + weightStyleOrder + weightStartGenders;
277
278 double score = 0;
279 score += ScoreSingleStarts * (weightSingleStarts / sumWeights);
280 score += ScoreSameStyleSequence * (weightSameStyleSequence / sumWeights);
281 score += ScorePersonStartPauses * (weightPersonStartPauses / sumWeights);
282 score += ScoreStyleOrder * (weightStyleOrder / sumWeights);
283 score += ScoreStartGenders * (weightStartGenders / sumWeights);
284
285 return LimitValue(score, 0, 100);
286 }

376 private double EvaluateSingleStarts()
377 {
378 if (Races.Count == 0) return 0;
379
380 int singleStarts = Races.Count(r => r.Starts.Count == 1);
381 double ratio = (double)singleStarts / Races.Count;
382
383 // The less single starts, the better
384 return 100 * (1 - ratio);

394 private double EvaluateSameStyleSequence()
395 {
396 if (Races.Count <= 1) return 100; // Only one race = no evaluation needed
397
398 int matchingPairs = 0;
399 for (int i = 1; i < Races.Count; i++)
400 {
401 if (Races[i].Style == Races[i - 1].Style)
402 matchingPairs++;
403 }
404
405 return 100 * ((double)matchingPairs / (Races.Count - 1));
406 }

416 private double EvaluatePersonStartPauses()
417 {
418 double penalty = 0;
419 double maxPenalty = 0; // This is calculated dynamically
420
421 Dictionary<Person, int> lastRaceIndex = new();
422 Dictionary<Person, double> individualPenalties = new();
423
424 int severelyAffectedPersons = 0; // Number of persons with hard penalty
425
426 for (int i = 0; i < Races.Count; i++)
427 {
428 foreach (var personStart in Races[i].Starts)
429 {
430 Person person = personStart.PersonObj;
431
432 if (lastRaceIndex.TryGetValue(person, out int lastIndex))
433 {
434 int distance = i - lastIndex;
435
436 double personPenalty = distance switch
437 {
438 1 => 100, // Extremly high penalty for directly consecutive starts
439 2 => 100, // Extremly high penalty for only one race pause
440 3 => 50, // Noticeable penalty for two races pause
441 4 => 30, // Noticeable penalty for three races pause
442 _ => Math.Max(0, 10 - distance) // Minimal penalty for 3+ races pause
443 };
444
445 // if the distance is below 3 races, this is regarded as severly affected
446 if (distance < 3)
447 {
448 severelyAffectedPersons++;
449 }
450
451 // Save the highest penalty for a person
452 if (!individualPenalties.ContainsKey(person) || individualPenalties[person] < personPenalty)
453 {
454 individualPenalties[person] = personPenalty;
455
456 maxPenalty += 100; // Add the highest possible penalty
457 }
458 }
459
460 lastRaceIndex[person] = i;
461 }
462 }
463
464 // Return the worst score if there are severely affected persons
465 if (severelyAffectedPersons > 0)
466 {
467 return 0;
468 }
469 else
470 {
471 // Sum up the hardest penalties per person
472 penalty = individualPenalties.Values.Sum();
473 double score = 100 - (penalty / maxPenalty * 100);
474 return LimitValue(score, 0, 100);
475 }
476 }

485 private double EvaluateStyleOrder()
486 {
487 if (Races.Count == 0) return 100;
488
489 // lower numbers should start earlier
490 Dictionary<SwimmingStyles, int> StylePriorities = new()
491 {
492 { SwimmingStyles.Breaststroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BREASTSTROKE) ?? 1 },
493 { SwimmingStyles.Freestyle, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_FREESTYLE) ?? 2 },
494 { SwimmingStyles.Backstroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BACKSTROKE) ?? 3 },
495 { SwimmingStyles.Butterfly, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BUTTERFLY) ?? 4 },
496 { SwimmingStyles.Medley, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_MEDLEY) ?? 5 },
497 { SwimmingStyles.WaterFlea, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_WATERFLEA) ?? 6 }
498 };
499
500 double penalty = 0;
501 foreach (var (race, index) in Races.Select((r, i) => (r, i)))
502 {
503 if (StylePriorities.TryGetValue(race.Style, out int priority))
504 {
505 int expectedMinIndex = Races.Count * priority / StylePriorities.Count;
506 if (index < expectedMinIndex)
507 {
508 penalty += (expectedMinIndex - index);
509 }
510 }
511 }
512
513 double maxPenalty = Races.Count * StylePriorities.Count;
514 return 100 - LimitValue(100 * (penalty / maxPenalty), 0, 100);
515 }

524 private double EvaluateStartGenders()
525 {
526 if (Races.Count == 0) return 100.0;
527
528 int totalRaces = Races.Count;
529 int homogeneousCount = 0;
530
531 foreach (Race race in Races)
532 {
533 List<Genders> genders = race.Starts.Select(s => s.PersonObj.Gender).Distinct().ToList();
534 if (genders.Count == 1)
535 {
536 homogeneousCount++;
537 }
538 }
539
540 double score = 100.0 * homogeneousCount / totalRaces;
541 return LimitValue(score, 0, 100);
542 }
