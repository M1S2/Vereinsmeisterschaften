464 private double EvaluatePersonStartPauses()
465 {
466 double penalty = 0;
467 double maxPenalty = 0; // This is calculated dynamically
468
469 Dictionary<Person, int> lastRaceIndex = new();
470 Dictionary<Person, double> individualPenalties = new();
471 List<Person> severelyAffectedPersons = new List<Person>();
472
473 uint shortPausesThreshold = _workspaceService?.Settings?.GetSettingValue<uint>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_SHORT_PAUSE_THRESHOLD) ?? 3;
474 int severelyAffectedPersonsCount = 0; // Number of persons with hard penalty
475
476 for (int i = 0; i < Races.Count; i++)
477 {
478 foreach (var personStart in Races[i].Starts)
479 {
480 // Only consider active starts
481 if (!personStart.IsActive) { continue; }
482
483 Person person = personStart.PersonObj;
484
485 if (lastRaceIndex.TryGetValue(person, out int lastIndex))
486 {
487 int distance = i - lastIndex;
488
489 double personPenalty = distance switch
490 {
491 1 => 100, // Extremly high penalty for directly consecutive starts
492 2 => 100, // Extremly high penalty for only one race pause
493 3 => 50, // Noticeable penalty for two races pause
494 4 => 30, // Noticeable penalty for three races pause
495 _ => Math.Max(0, 10 - distance) // Minimal penalty for 3+ races pause
496 };
497
498 // if the distance is below shortPausesThreshold races, this is regarded as severly affected
499 if (distance < shortPausesThreshold)
500 {
501 severelyAffectedPersonsCount++;
502 severelyAffectedPersons.Add(person);
503 }
504
505 // Save the highest penalty for a person
506 if (!individualPenalties.ContainsKey(person) || individualPenalties[person] < personPenalty)
507 {
508 individualPenalties[person] = personPenalty;
509
510 maxPenalty += 100; // Add the highest possible penalty
511 }
512 }
513
514 lastRaceIndex[person] = i;
515 }
516 }
517
518 PersonStartPausesSeverelyAffectedPersons = severelyAffectedPersons.Distinct().ToList();
519
520 // Return the worst score if there are severely affected persons
521 if (severelyAffectedPersonsCount > 0)
522 {
523 return 0;
524 }
525 else
526 {
527 // Sum up the hardest penalties per person
528 penalty = individualPenalties.Values.Sum();
529 double score = 100 - (penalty / maxPenalty * 100);
530 return LimitValue(score, 0, 100);
531 }
532 }

541 private double EvaluateStyleOrder()
542 {
543 if (Races.Count == 0) return 100;
544
545 // lower numbers should start earlier
546 Dictionary<SwimmingStyles, int> StylePriorities = new()
547 {
548 { SwimmingStyles.Breaststroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BREASTSTROKE) ?? 1 },
549 { SwimmingStyles.Freestyle, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_FREESTYLE) ?? 2 },
550 { SwimmingStyles.Backstroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BACKSTROKE) ?? 3 },
551 { SwimmingStyles.Butterfly, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BUTTERFLY) ?? 4 },
552 { SwimmingStyles.Medley, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_MEDLEY) ?? 5 },
553 { SwimmingStyles.WaterFlea, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_WATERFLEA) ?? 6 }
554 };
555
556 double penalty = 0;
557 foreach (var (race, index) in Races.Select((r, i) => (r, i)))
558 {
559 if (StylePriorities.TryGetValue(race.Style, out int priority))
560 {
561 int expectedMinIndex = Races.Count * priority / StylePriorities.Count;
562 if (index < expectedMinIndex)
563 {
564 penalty += (expectedMinIndex - index);
565 }
566 }
567 }
568
569 double maxPenalty = Races.Count * StylePriorities.Count;
570 return 100 - LimitValue(100 * (penalty / maxPenalty), 0, 100);
571 }

580 private double EvaluateStartGenders()
581 {
582 if (Races.Count == 0) return 100.0;
583
584 int totalRaces = Races.Count;
585 int homogeneousCount = 0;
586
587 foreach (Race race in Races)
588 {
589 List<Genders> genders = race.Starts.Select(s => s.PersonObj.Gender).Distinct().ToList();
590 if (genders.Count == 1)
591 {
592 homogeneousCount++;
593 }
594 }
595
596 double score = 100.0 * homogeneousCount / totalRaces;
597 return LimitValue(score, 0, 100);
598 }
