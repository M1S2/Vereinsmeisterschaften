273 public double Score
274 {
275 get
276 {
277 double weightSingleStarts = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SINGLE_STARTS) ?? 5;
278 double weightSameStyleSequence = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SAME_STYLE_SEQUENCE) ?? 5;
279 double weightPersonStartPauses = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_PERSON_START_PAUSES) ?? 65;
280 double weightStyleOrder = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_STYLE_ORDER) ?? 10;
281 double weightStartGenders = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_START_GENDERS) ?? 5;
282 double sumWeights = weightSingleStarts + weightSameStyleSequence + weightPersonStartPauses + weightStyleOrder + weightStartGenders;
283
284 double score = 0;
285 score += ScoreSingleStarts * (weightSingleStarts / sumWeights);
286 score += ScoreSameStyleSequence * (weightSameStyleSequence / sumWeights);
287 score += ScorePersonStartPauses * (weightPersonStartPauses / sumWeights);
288 score += ScoreStyleOrder * (weightStyleOrder / sumWeights);
289 score += ScoreStartGenders * (weightStartGenders / sumWeights);
290
291 return LimitValue(score, 0, 100);
292 }

382 private double EvaluateSingleStarts()
383 {
384 if (Races.Count == 0) return 0;

400 private double EvaluateSameStyleSequence()
401 {
402 if (Races.Count <= 1) return 100; // Only one race = no evaluation needed
403
404 int matchingPairs = 0;
405 for (int i = 1; i < Races.Count; i++)
406 {
407 if (Races[i].Style == Races[i - 1].Style)
408 matchingPairs++;
409 }
410
411 return 100 * ((double)matchingPairs / (Races.Count - 1));
412 }

422 private double EvaluatePersonStartPauses()
423 {
424 double penalty = 0;
425 double maxPenalty = 0; // This is calculated dynamically
426
427 Dictionary<Person, int> lastRaceIndex = new();
428 Dictionary<Person, double> individualPenalties = new();
429
430 int severelyAffectedPersons = 0; // Number of persons with hard penalty
431
432 for (int i = 0; i < Races.Count; i++)
433 {
434 foreach (var personStart in Races[i].Starts)
435 {
436 Person person = personStart.PersonObj;
437
438 if (lastRaceIndex.TryGetValue(person, out int lastIndex))
439 {
440 int distance = i - lastIndex;
441
442 double personPenalty = distance switch
443 {
444 1 => 100, // Extremly high penalty for directly consecutive starts
445 2 => 100, // Extremly high penalty for only one race pause
446 3 => 50, // Noticeable penalty for two races pause
447 4 => 30, // Noticeable penalty for three races pause
448 _ => Math.Max(0, 10 - distance) // Minimal penalty for 3+ races pause
449 };
450
451 // if the distance is below 3 races, this is regarded as severly affected
452 if (distance < 3)
453 {
454 severelyAffectedPersons++;
455 }
456
457 // Save the highest penalty for a person
458 if (!individualPenalties.ContainsKey(person) || individualPenalties[person] < personPenalty)
459 {
460 individualPenalties[person] = personPenalty;
461
462 maxPenalty += 100; // Add the highest possible penalty
463 }
464 }
465
466 lastRaceIndex[person] = i;
467 }
468 }
469
470 // Return the worst score if there are severely affected persons
471 if (severelyAffectedPersons > 0)
472 {
473 return 0;
474 }
475 else
476 {
477 // Sum up the hardest penalties per person
478 penalty = individualPenalties.Values.Sum();
479 double score = 100 - (penalty / maxPenalty * 100);
480 return LimitValue(score, 0, 100);
481 }
482 }

491 private double EvaluateStyleOrder()
492 {
493 if (Races.Count == 0) return 100;
494
495 // lower numbers should start earlier
496 Dictionary<SwimmingStyles, int> StylePriorities = new()
497 {
498 { SwimmingStyles.Breaststroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BREASTSTROKE) ?? 1 },
499 { SwimmingStyles.Freestyle, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_FREESTYLE) ?? 2 },
500 { SwimmingStyles.Backstroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BACKSTROKE) ?? 3 },
501 { SwimmingStyles.Butterfly, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BUTTERFLY) ?? 4 },
502 { SwimmingStyles.Medley, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_MEDLEY) ?? 5 },
503 { SwimmingStyles.WaterFlea, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_WATERFLEA) ?? 6 }
504 };
505
506 double penalty = 0;
507 foreach (var (race, index) in Races.Select((r, i) => (r, i)))
508 {
509 if (StylePriorities.TryGetValue(race.Style, out int priority))
510 {
511 int expectedMinIndex = Races.Count * priority / StylePriorities.Count;
512 if (index < expectedMinIndex)
513 {
514 penalty += (expectedMinIndex - index);
515 }
516 }
517 }
518
519 double maxPenalty = Races.Count * StylePriorities.Count;
520 return 100 - LimitValue(100 * (penalty / maxPenalty), 0, 100);
521 }

530 private double EvaluateStartGenders()
531 {
532 if (Races.Count == 0) return 100.0;
533
534 int totalRaces = Races.Count;
535 int homogeneousCount = 0;
536
537 foreach (Race race in Races)
538 {
539 List<Genders> genders = race.Starts.Select(s => s.PersonObj.Gender).Distinct().ToList();
540 if (genders.Count == 1)
541 {
542 homogeneousCount++;
543 }
544 }
545
546 double score = 100.0 * homogeneousCount / totalRaces;
547 return LimitValue(score, 0, 100);
548 }
