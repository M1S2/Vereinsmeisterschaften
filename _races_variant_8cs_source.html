277 public double Score
278 {
279 get
280 {
281 double weightSingleStarts = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SINGLE_STARTS) ?? 5;
282 double weightSameStyleSequence = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_SAME_STYLE_SEQUENCE) ?? 5;
283 double weightPersonStartPauses = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_PERSON_START_PAUSES) ?? 65;
284 double weightStyleOrder = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_STYLE_ORDER) ?? 10;
285 double weightStartGenders = _workspaceService?.Settings?.GetSettingValue<double>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_WEIGHT_START_GENDERS) ?? 5;
286 double sumWeights = weightSingleStarts + weightSameStyleSequence + weightPersonStartPauses + weightStyleOrder + weightStartGenders;
287
288 double score = 0;
289 score += ScoreSingleStarts * (weightSingleStarts / sumWeights);
290 score += ScoreSameStyleSequence * (weightSameStyleSequence / sumWeights);
291 score += ScorePersonStartPauses * (weightPersonStartPauses / sumWeights);
292 score += ScoreStyleOrder * (weightStyleOrder / sumWeights);
293 score += ScoreStartGenders * (weightStartGenders / sumWeights);
294
295 return LimitValue(score, 0, 100);
296 }

386 private double EvaluateSingleStarts()
387 {
388 if (Races.Count == 0) return 0;

404 private double EvaluateSameStyleSequence()
405 {
406 if (Races.Count <= 1) return 100; // Only one race = no evaluation needed
407
408 int matchingPairs = 0;
409 for (int i = 1; i < Races.Count; i++)
410 {
411 if (Races[i].Style == Races[i - 1].Style)
412 matchingPairs++;
413 }
414
415 return 100 * ((double)matchingPairs / (Races.Count - 1));
416 }

426 private double EvaluatePersonStartPauses()
427 {
428 double penalty = 0;
429 double maxPenalty = 0; // This is calculated dynamically
430
431 Dictionary<Person, int> lastRaceIndex = new();
432 Dictionary<Person, double> individualPenalties = new();
433
434 int severelyAffectedPersons = 0; // Number of persons with hard penalty
435
436 for (int i = 0; i < Races.Count; i++)
437 {
438 foreach (var personStart in Races[i].Starts)
439 {
440 Person person = personStart.PersonObj;
441
442 if (lastRaceIndex.TryGetValue(person, out int lastIndex))
443 {
444 int distance = i - lastIndex;
445
446 double personPenalty = distance switch
447 {
448 1 => 100, // Extremly high penalty for directly consecutive starts
449 2 => 100, // Extremly high penalty for only one race pause
450 3 => 50, // Noticeable penalty for two races pause
451 4 => 30, // Noticeable penalty for three races pause
452 _ => Math.Max(0, 10 - distance) // Minimal penalty for 3+ races pause
453 };
454
455 // if the distance is below 3 races, this is regarded as severly affected
456 if (distance < 3)
457 {
458 severelyAffectedPersons++;
459 }
460
461 // Save the highest penalty for a person
462 if (!individualPenalties.ContainsKey(person) || individualPenalties[person] < personPenalty)
463 {
464 individualPenalties[person] = personPenalty;
465
466 maxPenalty += 100; // Add the highest possible penalty
467 }
468 }
469
470 lastRaceIndex[person] = i;
471 }
472 }
473
474 // Return the worst score if there are severely affected persons
475 if (severelyAffectedPersons > 0)
476 {
477 return 0;
478 }
479 else
480 {
481 // Sum up the hardest penalties per person
482 penalty = individualPenalties.Values.Sum();
483 double score = 100 - (penalty / maxPenalty * 100);
484 return LimitValue(score, 0, 100);
485 }
486 }

495 private double EvaluateStyleOrder()
496 {
497 if (Races.Count == 0) return 100;
498
499 // lower numbers should start earlier
500 Dictionary<SwimmingStyles, int> StylePriorities = new()
501 {
502 { SwimmingStyles.Breaststroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BREASTSTROKE) ?? 1 },
503 { SwimmingStyles.Freestyle, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_FREESTYLE) ?? 2 },
504 { SwimmingStyles.Backstroke, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BACKSTROKE) ?? 3 },
505 { SwimmingStyles.Butterfly, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_BUTTERFLY) ?? 4 },
506 { SwimmingStyles.Medley, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_MEDLEY) ?? 5 },
507 { SwimmingStyles.WaterFlea, _workspaceService?.Settings?.GetSettingValue<int>(WorkspaceSettings.GROUP_RACE_CALCULATION, WorkspaceSettings.SETTING_RACE_CALCULATION_PRIORITY_STYLE_WATERFLEA) ?? 6 }
508 };
509
510 double penalty = 0;
511 foreach (var (race, index) in Races.Select((r, i) => (r, i)))
512 {
513 if (StylePriorities.TryGetValue(race.Style, out int priority))
514 {
515 int expectedMinIndex = Races.Count * priority / StylePriorities.Count;
516 if (index < expectedMinIndex)
517 {
518 penalty += (expectedMinIndex - index);
519 }
520 }
521 }
522
523 double maxPenalty = Races.Count * StylePriorities.Count;
524 return 100 - LimitValue(100 * (penalty / maxPenalty), 0, 100);
525 }

534 private double EvaluateStartGenders()
535 {
536 if (Races.Count == 0) return 100.0;
537
538 int totalRaces = Races.Count;
539 int homogeneousCount = 0;
540
541 foreach (Race race in Races)
542 {
543 List<Genders> genders = race.Starts.Select(s => s.PersonObj.Gender).Distinct().ToList();
544 if (genders.Count == 1)
545 {
546 homogeneousCount++;
547 }
548 }
549
550 double score = 100.0 * homogeneousCount / totalRaces;
551 return LimitValue(score, 0, 100);
552 }
