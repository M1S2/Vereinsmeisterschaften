335 private double EvaluateSingleStarts()
336 {
337 if (Races.Count == 0) return 0;

353 private double EvaluateSameStyleSequence()
354 {
355 if (Races.Count <= 1) return 100; // Only one race = no evaluation needed
356
357 int matchingPairs = 0;
358 for (int i = 1; i < Races.Count; i++)
359 {
360 if (Races[i].Style == Races[i - 1].Style)
361 matchingPairs++;
362 }
363
364 return 100 * ((double)matchingPairs / (Races.Count - 1));
365 }

375 private double EvaluatePersonStartPauses()
376 {
377 double penalty = 0;
378 double maxPenalty = 0; // This is calculated dynamically
379
380 Dictionary<Person, int> lastRaceIndex = new();
381 Dictionary<Person, double> individualPenalties = new();
382
383 int severelyAffectedPersons = 0; // Number of persons with hard penalty
384
385 for (int i = 0; i < Races.Count; i++)
386 {
387 foreach (var personStart in Races[i].Starts)
388 {
389 Person person = personStart.PersonObj;
390
391 if (lastRaceIndex.TryGetValue(person, out int lastIndex))
392 {
393 int distance = i - lastIndex;
394
395 double personPenalty = distance switch
396 {
397 1 => 100, // Extremly high penalty for directly consecutive starts
398 2 => 100, // Extremly high penalty for only one race pause
399 3 => 50, // Noticeable penalty for two races pause
400 4 => 30, // Noticeable penalty for three races pause
401 _ => Math.Max(0, 10 - distance) // Minimal penalty for 3+ races pause
402 };
403
404 // if the distance is below 3 races, this is regarded as severly affected
405 if (distance < 3)
406 {
407 severelyAffectedPersons++;
408 }
409
410 // Save the highest penalty for a person
411 if (!individualPenalties.ContainsKey(person) || individualPenalties[person] < personPenalty)
412 {
413 individualPenalties[person] = personPenalty;
414
415 maxPenalty += 100; // Add the highest possible penalty
416 }
417 }
418
419 lastRaceIndex[person] = i;
420 }
421 }
422
423 // Return the worst score if there are severely affected persons
424 if (severelyAffectedPersons > 0)
425 {
426 return 0;
427 }
428 else
429 {
430 // Sum up the hardest penalties per person
431 penalty = individualPenalties.Values.Sum();
432 double score = 100 - (penalty / maxPenalty * 100);
433 return LimitValue(score, 0, 100);
434 }
435 }

454 private double EvaluateStyleOrder()
455 {
456 if (Races.Count == 0) return 100;
457
458 double penalty = 0;
459 foreach (var (race, index) in Races.Select((r, i) => (r, i)))
460 {
461 if (STYLE_PRIORITY.TryGetValue(race.Style, out int priority))
462 {
463 int expectedMinIndex = Races.Count * priority / STYLE_PRIORITY.Count;
464 if (index < expectedMinIndex)
465 {
466 penalty += (expectedMinIndex - index);
467 }
468 }
469 }
470
471 double maxPenalty = Races.Count * STYLE_PRIORITY.Count;
472 return 100 - LimitValue(100 * (penalty / maxPenalty), 0, 100);
473 }

482 private double EvaluateStartGenders()
483 {
484 if (Races.Count == 0) return 100.0;
485
486 int totalRaces = Races.Count;
487 int homogeneousCount = 0;
488
489 foreach (Race race in Races)
490 {
491 List<Genders> genders = race.Starts.Select(s => s.PersonObj.Gender).Distinct().ToList();
492 if (genders.Count == 1)
493 {
494 homogeneousCount++;
495 }
496 }
497
498 double score = 100.0 * homogeneousCount / totalRaces;
499 return LimitValue(score, 0, 100);
500 }
