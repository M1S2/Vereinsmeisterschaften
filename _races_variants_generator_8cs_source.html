45 public async Task<List<RacesVariant>> GenerateBestRacesAsync(List<List<PersonStart>> sets, CancellationToken cancellationToken = default)
46 {
47 ConcurrentBag<RacesVariant> bestRaces = new ConcurrentBag<RacesVariant>();
48 int attempts = 0;
49 int foundVariants = 0;
50
51 await Task.Run(() =>
52 {
53 Parallel.For(0, _maxIterations, new ParallelOptions() { MaxDegreeOfParallelism = (int)(0.5 * Environment.ProcessorCount) }, (i, state) =>
54 {
55 if (cancellationToken.IsCancellationRequested || bestRaces.Count >= _requiredVariantsCount)
56 {
57 state.Stop();
58 return;
59 }
60
61 // Random grouping creation
62 RacesVariant candidate = CreateValidGrouping(sets);
63
64 // Calculate score and keep if above threshold
65 double score = candidate.CalculateScore();
66 if (score >= _minScoreThreshold)
67 {
68 bestRaces.Add(candidate);
69 Interlocked.Increment(ref foundVariants);
70 }
71
72 Interlocked.Increment(ref attempts);
73
74 // Calculate progress as a mixture of iteration progress and solution progress
75 double iterationProgress = (double)attempts / _maxIterations;
76 double solutionProgress = (double)foundVariants / _requiredVariantsCount;
77 double overallProgress = Math.Min(1.0, Math.Max(iterationProgress, solutionProgress));
78
79 _progress?.Report(overallProgress * 100);
80 });
81 }, cancellationToken);
82
83 return bestRaces.OrderByDescending(r => r.Score).Take(_requiredVariantsCount).ToList();

91 private RacesVariant CreateValidGrouping(List<List<PersonStart>> sets)
92 {
93 List<List<PersonStart>> groups = new List<List<PersonStart>>();
94 List<List<PersonStart>> shuffledSets = sets.OrderBy(_ => _random.Next()).ToList(); // Shuffle sets for more randomness
95 int oneElementGroupsCount = 0;
96
97 foreach (List<PersonStart> set in shuffledSets)
98 {
99 HashSet<PersonStart> remainingElements = new HashSet<PersonStart>(set.OrderBy(_ => _random.Next())); // Shuffle each set
100
101 while (remainingElements.Count > 0)
102 {
103 int maxSize = Math.Min(_maxGroupSize, remainingElements.Count);
104 int minSize = Math.Max(2, (int)(maxSize * 0.5)); // Avoid too small groups
105
106 int groupSize = _random.Next(minSize, maxSize + 1);
107 List<PersonStart> selectedGroup = remainingElements.Take(groupSize).ToList();
108
109 if (selectedGroup.Count == 1)
110 {
111 oneElementGroupsCount++;
112 }
113
114 groups.Add(selectedGroup);
115 foreach (PersonStart item in selectedGroup)
116 remainingElements.Remove(item);
117 }
118 }
119
120 // If too many single element groups were created → Adjust instead of restart
121 if ((double)oneElementGroupsCount / groups.Count > _maxOneElementGroupPercentage)
122 {
123 groups = MergeSmallGroups(groups, sets);
124 }
125
126 RacesVariant variant = new RacesVariant();
127 foreach (var group in groups)
128 {
129 variant.Races.Add(new Race(group));
130 }
131
132 return variant;
133 }

141 private List<List<PersonStart>> MergeSmallGroups(List<List<PersonStart>> groups, List<List<PersonStart>> sets)
142 {
143 List<List<PersonStart>> result = new();
144 HashSet<int> mergedIndices = new();
145
146 for (int i = 0; i < groups.Count; i++)
147 {
148 if (mergedIndices.Contains(i))
149 continue;
150
151 var groupA = groups[i];
152
153 // Nur kleine Gruppen betrachten
154 if (groupA.Count >= 1)
155 {
156 result.Add(groupA);
157 continue;
158 }
159
160 bool merged = false;
161
162 for (int j = i + 1; j < groups.Count; j++)
163 {
164 if (mergedIndices.Contains(j))
165 continue;
166
167 var groupB = groups[j];
168
169 if (groupA.Count + groupB.Count <= _maxGroupSize &&
170 AreFromSameSet(groupA.Concat(groupB).ToList(), sets))
171 {
172 result.Add(groupA.Concat(groupB).ToList());
173 mergedIndices.Add(i);
174 mergedIndices.Add(j);
175 merged = true;
176 break;
177 }
178 }
179
180 if (!merged)
181 {
182 result.Add(groupA);
183 }
184 }
185
