37 public void UpdateScoresForPerson(Person person)
38 {
39 _competitionService.UpdateAllCompetitionsForPerson(person);
40
41 foreach (PersonStart start in person?.Starts?.Values)
42 {
43 if (start == null) { continue; }
44
45 Competition competition = start.CompetitionObj;
46 if (competition == null) { continue; }
47 // If the start time equals the competition best time the score will be 100
48 // If the person swims faster, the score is higher
49 if (start.Time.TotalMilliseconds == 0)
50 {
51 start.Score = 0;
52 }
53 else
54 {
55 // Original formula from old Vereinsmeisterschaften tool:
56 // score = 100 * (1 - ((start_time - competition_best_time) / competition_best_time)) = 100 * (2 - (start_time / competition_best_time))
57 // if startTime == bestTime => BEST_TIME_SCORE
58 // linear equation around this point
59 // zero points if startTime >= 2* bestTime
60 start.Score = (2 - (start.Time.TotalMilliseconds / competition.BestTime.TotalMilliseconds)) * BEST_TIME_SCORE;
61
62 // Limit score to 0
63 if (start.Score < 0) { start.Score = 0; }
64 }
65 }
66 }

87
88 // Group all starts by the score. It is possible to have more than one start with the same score leading to the same podium place
89 List<IGrouping<double, PersonStart>> groupedStarts = bestStarts.GroupBy(s => s.Score).ToList();

102 public List<Person> GetPersonsSortedByScore(ResultTypes resultType)

125 public List<PersonStart> GetWinnersPodiumStarts(ResultTypes resultType, ResultPodiumsPlaces podiumsPlace)

134 else

138 }
139
140 // Group all starts by the score. It is possible to have more than one start with the same score leading to the same podium place
141 List<IGrouping<double, PersonStart>> groupedStarts = bestStarts.GroupBy(s => s.Score).ToList();
142 if(podiumsPlace == ResultPodiumsPlaces.Gold && groupedStarts.Count > 0)
143 {
144 return groupedStarts[0].ToList();
145 }
146 else if (podiumsPlace == ResultPodiumsPlaces.Silver && groupedStarts.Count > 1)
147 {
148 return groupedStarts[1].ToList();
149 }
150 else if (podiumsPlace == ResultPodiumsPlaces.Bronze && groupedStarts.Count > 2)
