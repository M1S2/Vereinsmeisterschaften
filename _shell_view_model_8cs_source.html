168 private void _workspaceService_PropertyChanged(object sender, System.ComponentModel.PropertyChangedEventArgs e)
169 {
170 switch (e.PropertyName)
171 {
172 case nameof(IWorkspaceService.HasUnsavedChanges): OnPropertyChanged(nameof(HasUnsavedChanges)); break;
173 case nameof(IWorkspaceService.PersistentPath): OnPropertyChanged(nameof(CurrentWorkspaceFolder)); break;
174 case nameof(IWorkspaceService.IsWorkspaceOpen):
175 {
176 OnPropertyChanged(nameof(CurrentWorkspaceFolder));
177 updateMenuItemsEnabledState();
178 break;
179 }
180 default: break;
181 }

203 // Close behaviour: https://github.com/MahApps/MahApps.Metro/issues/3535
204 private bool _forceClose = false;
205
209 public event EventHandler WindowCloseRequested;
210 protected void OnClosing(CancelEventArgs e)
211 {
212 // force method to abort - we'll force a close explicitly
213 e.Cancel = true;
214
215 if (_forceClose)
216 {
217 // cleanup code already ran
218 e.Cancel = false;
219 return;
220 }
221
222 // execute shutdown logic - set ForceClose and request Close() to actually close the window
223 Dispatcher.CurrentDispatcher.InvokeAsync(async () =>
224 {
225 bool save = await checkForUnsavedChangesAndQueryUserAction();
226 if (save)
227 {
228 try
229 {
230 await _workspaceService.Save(CancellationToken.None);
231 }
232 catch (Exception ex)
233 {
234 await _dialogCoordinator.ShowMessageAsync(this, Resources.ErrorString, ex.Message);
235 }
236 }
237 App.Current.Properties["LastWorkspaceFolder"] = CurrentWorkspaceFolder;
238 _forceClose = true;
239 WindowCloseRequested?.Invoke(this, null); // Notify the ShellWindow to close
240 }, DispatcherPriority.Normal);
241 }

247 private async Task<bool> checkForUnsavedChangesAndQueryUserAction()
248 {
249 bool save = false;
250 if (_workspaceService?.HasUnsavedChanges ?? false)
251 {
252 MetroDialogSettings dialogButtonSettings = new MetroDialogSettings()
253 {
254 AffirmativeButtonText = Resources.SaveString,
255 NegativeButtonText = Resources.DontSaveString,
256 DefaultButtonFocus = MessageDialogResult.Affirmative
257 };
258 MessageDialogResult dialogResult = await _dialogCoordinator.ShowMessageAsync(this, Resources.UnsavedChangesString, Resources.UnsavedChangesSavePromptString,
259 MessageDialogStyle.AffirmativeAndNegative, dialogButtonSettings);
260 switch (dialogResult)
261 {
262 case MessageDialogResult.Affirmative: save = true; break;
263 case MessageDialogResult.Negative: save = false; break;
264 default: break;
265 }
266 }
267 return save;
268 }
