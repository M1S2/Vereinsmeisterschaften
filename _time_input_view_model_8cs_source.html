106 public Predicate<object> AvailablePersonStartsFilterPredicate
107 {
108 get
109 {
110 return (item) =>
111 {
112 PersonStart personStart = item as PersonStart;
113
114 bool filterResult = true;
115 switch(FilterPersonStartMode)
116 {
117 case TimeInputPersonStartFilterModes.Person:
118 return personStart?.PersonObj == FilteredPerson;
119 case TimeInputPersonStartFilterModes.RaceID:
120 Race race = PersistedRacesVariant?.Races?.Where(r => r.Starts.Contains(personStart)).FirstOrDefault();
121 return race == null ? false : race.RaceID == FilteredRaceID;
122 case TimeInputPersonStartFilterModes.CompetitionID:
123 return (personStart?.CompetitionObj?.ID ?? -1) == FilteredCompetitionID;
124 case TimeInputPersonStartFilterModes.None:
125 default:
126 break;
127 }
128 return filterResult;
129 };
130 }
