131 public Predicate<object> AvailablePersonStartsFilterPredicate
132 {
133 get
134 {
135 return (item) =>
136 {
137 PersonStart personStart = item as PersonStart;
138
139 bool filterResult = true;
140 switch(FilterPersonStartMode)
141 {
142 case TimeInputPersonStartFilterModes.Person:
143 return personStart?.PersonObj == FilteredPerson;
144 case TimeInputPersonStartFilterModes.RaceID:
145 Race race = PersistedRacesVariant?.Races?.Where(r => r.Starts.Contains(personStart)).FirstOrDefault();
146 return race == null ? false : race.RaceID == FilteredRaceID;
147 case TimeInputPersonStartFilterModes.CompetitionID:
148 return (personStart?.CompetitionObj?.Id ?? -1) == FilteredCompetitionID;
149 case TimeInputPersonStartFilterModes.None:
150 default:
151 break;
152 }
153 return filterResult;
154 };
155 }
