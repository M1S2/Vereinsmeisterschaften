201 #region Digit Update Logic
202
211 private void handleDigitInput(byte digit)
212 {
213 if(digit > 9) { digit = 9; }
214
215 int oldValueMultiplier = _digitEntry_StartWithFirstDigit ? 0 : 10;
216 _digitEntry_StartWithFirstDigit = false;
217 if(digit == 0) { _digitEntry_CountEnteredZeroes++; }
218 switch (EditMode)
219 {
220 case TimeSpanControlEditModes.EditHours:
221 Hours = (Hours * oldValueMultiplier) + digit;
222 if (Hours > 2 || _digitEntry_CountEnteredZeroes >= (digit == 0 ? 2 : 1)) { EditMode = TimeSpanControlEditModes.EditMinutes; }
223 break;
224 case TimeSpanControlEditModes.EditMinutes:
225 Minutes = (Minutes * oldValueMultiplier) + digit;
226 if (Minutes >= 6 || _digitEntry_CountEnteredZeroes >= (digit == 0 ? 2 : 1)) { EditMode = TimeSpanControlEditModes.EditSeconds; }
227 break;
228 case TimeSpanControlEditModes.EditSeconds:
229 Seconds = (Seconds * oldValueMultiplier) + digit;
230 if(MillisecondsVisible)
231 {
232 if (Seconds >= 6 || _digitEntry_CountEnteredZeroes >= (digit == 0 ? 2 : 1)) { EditMode = TimeSpanControlEditModes.EditMilliseconds; }
233 }
234 else
235 {
236 if (Seconds * 10 >= 59 || _digitEntry_CountEnteredZeroes >= 2) { _digitEntry_StartWithFirstDigit = true; _digitEntry_CountEnteredZeroes = 0; }
237 }
238 break;
239 case TimeSpanControlEditModes.EditMilliseconds:
240 Milliseconds = (Milliseconds * oldValueMultiplier) + digit;
241 if (Milliseconds * 10 >= 999 || _digitEntry_CountEnteredZeroes >= 3) { _digitEntry_StartWithFirstDigit = true; _digitEntry_CountEnteredZeroes = 0; }
242 break;
243 }
244 }
245
249 private void increaseDigit()
250 {
251 switch (EditMode)
252 {
253 case TimeSpanControlEditModes.EditHours:
254 if(Hours < 23) { Hours++; }
255 break;
256 case TimeSpanControlEditModes.EditMinutes:
257 if(Minutes < 59) { Minutes++; }
258 break;
259 case TimeSpanControlEditModes.EditSeconds:
260 if(Seconds < 59) { Seconds++; }
261 break;
262 case TimeSpanControlEditModes.EditMilliseconds:
263 if(Milliseconds < 999) { Milliseconds++; }
264 break;
265 }
266 }

273 switch (EditMode)

281 case TimeSpanControlEditModes.EditSeconds:
282 if (Seconds > 0) { Seconds--; }
283 break;
284 case TimeSpanControlEditModes.EditMilliseconds:
285 if (Milliseconds > 0) { Milliseconds--; }
286 break;
287 }

324 private void UserControl_PreviewKeyDown(object sender, KeyEventArgs e)
325 {
326 if (e.Key == Key.Tab)
327 {
328 // Return here to enable the default behavior of the Tab key
329 return;
330 }
331 else if ((e.Key >= Key.D0 && e.Key <= Key.D9) || (e.Key >= Key.NumPad0 && e.Key <= Key.NumPad9))
332 {
333 string keyString = e.Key.ToString().Replace("NumPad", "").Replace("D", "");
334 byte digit;
335 if (byte.TryParse(keyString, out digit))
336 {
337 handleDigitInput(digit);
338 }
339 }
340 else if (e.Key == Key.Right)
341 {
342 TimeSpanControlEditModes highestMode = TimeSpanControlEditModes.EditMilliseconds;
343 if(!MillisecondsVisible) { highestMode = TimeSpanControlEditModes.EditSeconds; }
344 if (EditMode < highestMode)
345 {
346 EditMode++;
347 }
348 }
349 else if (e.Key == Key.Left)
350 {
351 TimeSpanControlEditModes lowestMode = TimeSpanControlEditModes.EditHours;
352 if (!HoursVisible) { lowestMode = TimeSpanControlEditModes.EditMinutes; }
353 if (EditMode > lowestMode)
354 {
355 EditMode--;
356 }
357 }
358 else if(e.Key == Key.Up)
359 {
360 increaseDigit();
361 }
362 else if (e.Key == Key.Down)
363 {
364 decreaseDigit();
365 }
366 e.Handled = true;
367 }
