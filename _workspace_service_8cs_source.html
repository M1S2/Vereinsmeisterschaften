251 public async Task<bool> Load(string path, CancellationToken cancellationToken)
252 {
253 if (!Directory.Exists(path)) { return false; }
254
255 if(IsWorkspaceOpen)
256 {
257 await CloseWorkspace(cancellationToken, true);
258 }
259
260 string previousPath = PersistentPath;
261 PersistentPath = path;
262
263 bool openResult = false;
264 Exception exception = null;
265 try
266 {
267 // Workspace settings
268 Settings = new WorkspaceSettings();
269 Settings.Load(_fileService, WorkspaceSettingsFilePath);
270 Settings.PropertyChanged += Settings_PropertyChanged;
271
272 // Persons
273 openResult = await _personService.Load(PersonFilePath, cancellationToken);
274 if(!openResult) { return openResult; }
275
276 // Competitions
277 openResult = await _competitionService.Load(CompetitionsFilePath, cancellationToken);
278 if (!openResult) { return openResult; }
279
280 _competitionService.UpdateAllCompetitionsForPerson();
281
282 // Best Race
283 openResult = await _raceService.Load(BestRaceFilePath, cancellationToken);
284
285 SettingsPersistedInFile = new WorkspaceSettings(Settings);
286 OnPropertyChangedAllHasUnsavedChanges();
287 IsWorkspaceOpen = openResult;
288 }
289 catch(Exception ex)
290 {
291 PersistentPath = previousPath;
292 OnPropertyChangedAllHasUnsavedChanges();
293 OnPropertyChanged(nameof(IsWorkspaceOpen));
294 exception = ex;
295 }
296 OnFileFinished?.Invoke(this, null);
297 if (exception != null) { throw exception; }
298 return openResult;
299 }

309 public async Task<bool> Save(CancellationToken cancellationToken, string path = "")
310 {

313 bool saveResult = false;

322 if (!saveResult) { return saveResult; }
323
324 // Competitions
325 saveResult = await _competitionService.Save(cancellationToken, CompetitionsFilePath);
326
327 // Best Race
328 saveResult = await _raceService.Save(cancellationToken, BestRaceFilePath);
329
330 SettingsPersistedInFile = new WorkspaceSettings(Settings);
331 OnPropertyChangedAllHasUnsavedChanges();
332 }
333 catch (Exception ex)
334 {
335 OnPropertyChangedAllHasUnsavedChanges();
336 exception = ex;
337 }
338 OnFileFinished?.Invoke(this, null);
339 if (exception != null) { throw exception; }
340 return saveResult;
341 }
