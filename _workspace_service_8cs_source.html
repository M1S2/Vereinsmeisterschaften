218 public async Task<bool> Load(string path, CancellationToken cancellationToken)
219 {
220 if (!Directory.Exists(path)) { return false; }
221
222 if(IsWorkspaceOpen)
223 {
224 await CloseWorkspace(cancellationToken, true);
225 }
226
227 string previousPath = PersistentPath;
228 PersistentPath = path;
229
230 bool openResult = false;
231 Exception exception = null;
232 try
233 {
234 // Workspace settings
235 Settings = new WorkspaceSettings();
236 Settings.Load(_fileService, WorkspaceSettingsFilePath);
237 Settings.PropertyChanged += Settings_PropertyChanged;
238
239 // Competitions
240 openResult = await _competitionService.Load(getFilePathToLoadFrom(KEY_FILENAME_COMPETITIONS), cancellationToken);
241 if (!openResult) { return openResult; }
242
243 // Persons
244 openResult = await _personService.Load(getFilePathToLoadFrom(KEY_FILENAME_PERSON), cancellationToken);
245 if(!openResult) { return openResult; }
246
247 _competitionService.UpdateAllCompetitionsForPerson();
248
249 // Best Race
250 openResult = await _raceService.Load(getFilePathToLoadFrom(KEY_FILENAME_BESTRACE), cancellationToken);
251
252 SettingsPersistedInFile = new WorkspaceSettings(Settings);
253 OnPropertyChangedAllHasUnsavedChanges();
254 IsWorkspaceOpen = openResult;
255 }
256 catch(Exception ex)
257 {
258 PersistentPath = previousPath;
259 OnPropertyChangedAllHasUnsavedChanges();
260 OnPropertyChanged(nameof(IsWorkspaceOpen));
261 exception = ex;
262 }
263 OnFileFinished?.Invoke(this, null);
264 if (exception != null) { throw exception; }
265 return openResult;
266 }

276 public async Task<bool> Save(CancellationToken cancellationToken, string path = "")
277 {
278 if (string.IsNullOrEmpty(path)) { path = PersistentPath; }
279
280 bool saveResult = false;
281 Exception exception = null;
282 try
283 {
284 // Workspace settings
285 Settings?.Save(_fileService, WorkspaceSettingsFilePath);
286
287 // Competitions (delete old file if the file name changed)
288 if (!await saveServiceAsync(KEY_FILENAME_COMPETITIONS, _competitionService, cancellationToken))
289 return false;
290
291 // Persons
292 if (!await saveServiceAsync(KEY_FILENAME_PERSON, _personService, cancellationToken))
293 return false;
294
295 // Best Race
296 if (!await saveServiceAsync(KEY_FILENAME_BESTRACE, _raceService, cancellationToken))
297 return false;
298
299 SettingsPersistedInFile = new WorkspaceSettings(Settings);
300 OnPropertyChangedAllHasUnsavedChanges();
301 }
302 catch (Exception ex)
303 {
304 OnPropertyChangedAllHasUnsavedChanges();
305 exception = ex;
306 }
307 OnFileFinished?.Invoke(this, null);
308 if (exception != null) { throw exception; }
309 return saveResult;

319 private async Task<bool> saveServiceAsync(string fileNameResourceKey, ISaveable saveableService, CancellationToken cancellationToken)
320 {
321 string newPath = getFilePathToSaveTo(fileNameResourceKey);
322
323 if (!string.IsNullOrEmpty(saveableService.PersistentPath) &&
324 File.Exists(saveableService.PersistentPath) &&
325 !string.Equals(saveableService.PersistentPath, newPath, StringComparison.OrdinalIgnoreCase))
326 {
327 File.Delete(saveableService.PersistentPath);
328 }
329
330 return await saveableService.Save(cancellationToken, newPath);
331 }

380 private string getLocalizedFilePath(string resourceKey, CultureInfo culture = null)
381 {
382 string localizedFileName = _fileNameResources.GetString(resourceKey, culture == null ? CultureInfo.CurrentUICulture : culture)
383 ?? _fileNameResources.GetString(resourceKey, CultureInfo.InvariantCulture);

400 private string getFilePathToLoadFrom(string resourceKey)
401 {
402 if (string.IsNullOrEmpty(resourceKey)) { return string.Empty; }
403 List<string> candidates = new List<string>();
404
405 // Test if the file for the current culture exists
406 string current = getLocalizedFilePath(resourceKey);
407 if (File.Exists(current))
408 {
409 return current;
410 }
411
412 // Check all known cultures
413 List<string> localizedFileNames = GeneralLocalizationHelper.GetAllTranslationsForKey(_fileNameResources, resourceKey).Values.ToList();
414 List<string> localizedFilePaths = localizedFileNames.Select(f => Path.Combine(PersistentPath, f)).ToList();
415 foreach (string localizedFilePath in localizedFilePaths)
416 {
417 if (File.Exists(localizedFilePath) && !candidates.Contains(localizedFilePath, StringComparer.OrdinalIgnoreCase))
418 {
419 candidates.Add(localizedFilePath);
420 }
421 }
422
423 // Use newest file depending on last write time
424 return candidates.Select(f => new FileInfo(f))
425 .OrderByDescending(f => f.LastWriteTimeUtc)
426 .FirstOrDefault()?.FullName;
427 }
