130 public void AddGroup(WorkspaceSettingsGroup group)
131 {
132 if (Groups.Where(g => string.Compare(g.GroupKey, group.GroupKey, true) == 0).Any())
133 {
134 // A group with the same key already exists
135 throw new ArgumentException($"A group with the key '{group.GroupKey}' already exists.", nameof(group));
136 }
137 else
138 {
139 Groups.Add(group);
140 group.PropertyChanged += (sender, e) =>
141 {
142 switch (e.PropertyName)
143 {
144 case nameof(WorkspaceSettingsGroup.HasChanged): OnPropertyChanged(nameof(HasChanged)); break;
145 default: break;
146 }
147 };
148 }
149 }
150
157 public WorkspaceSettingsGroup GetGroup(string groupKey, bool createGroupIfNotExisting = false)
158 {
159 WorkspaceSettingsGroup group = Groups.FirstOrDefault(g => string.Compare(g.GroupKey, groupKey, true) == 0);
160 if (group == null && createGroupIfNotExisting)
161 {
162 group = new WorkspaceSettingsGroup(groupKey);
163 AddGroup(group);
164 }
165 return group;
166 }

304 public void Save(IFileService fileService, string filePath)
305 {
306 CreateSnapshot();
307 List<SerializableWorkspaceSettingsGroup> serializableGroups = Groups.Select(group => new SerializableWorkspaceSettingsGroup
308 {
309 GroupKey = group.GroupKey,
310 Settings = group.Settings.Select(s => new SerializableWorkspaceSetting
311 {
312 Key = s.Key,
313 Value = s.UntypedValue
314 }).ToList()
315 }).ToList();
316
317 fileService.Save(Path.GetDirectoryName(filePath), Path.GetFileName(filePath), serializableGroups);
318 }
319
327 public void Load(IFileService fileService, string filePath)
328 {
329 List<SerializableWorkspaceSettingsGroup> serializableGroups = fileService.Read<List<SerializableWorkspaceSettingsGroup>>(Path.GetDirectoryName(filePath), Path.GetFileName(filePath));
330 Groups.Clear();
331 initializeSettings();
332 if (serializableGroups != null)
333 {
334 foreach (SerializableWorkspaceSettingsGroup serializableGroup in serializableGroups)
335 {
336 foreach (SerializableWorkspaceSetting serializableSetting in serializableGroup.Settings)
337 {
338 IWorkspaceSetting setting = GetSetting(serializableGroup.GroupKey, serializableSetting.Key);
339 if (setting != null)
340 {
341 object serializableValue = serializableSetting.Value;
342 if(serializableValue is string strValue && setting.ValueType.IsEnum)
343 {
344 serializableValue = Enum.Parse(setting.ValueType, strValue);
345 }
346 setting.UntypedValue = serializableValue;
347 }
348 }
349 }
350 }
351 CreateSnapshot();
352 }
