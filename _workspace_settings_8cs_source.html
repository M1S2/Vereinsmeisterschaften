140 public void AddGroup(WorkspaceSettingsGroup group)
141 {
142 if (Groups.Where(g => string.Compare(g.GroupKey, group.GroupKey, true) == 0).Any())
143 {
144 // A group with the same key already exists
145 throw new ArgumentException($"A group with the key '{group.GroupKey}' already exists.", nameof(group));
146 }
147 else
148 {
149 Groups.Add(group);
150 group.PropertyChanged += (sender, e) =>
151 {
152 switch (e.PropertyName)
153 {
154 case nameof(WorkspaceSettingsGroup.HasChanged): OnPropertyChanged(nameof(HasChanged)); break;
155 default: break;
156 }
157 };
158 }
159 }

167 public WorkspaceSettingsGroup GetGroup(string groupKey, bool createGroupIfNotExisting = false)
168 {
169 WorkspaceSettingsGroup group = Groups.FirstOrDefault(g => string.Compare(g.GroupKey, groupKey, true) == 0);
170 if (group == null && createGroupIfNotExisting)
171 {
172 group = new WorkspaceSettingsGroup(groupKey);
173 AddGroup(group);
174 }
175 return group;
176 }

314 public void Save(IFileService fileService, string filePath)
315 {
316 CreateSnapshot();
317 List<SerializableWorkspaceSettingsGroup> serializableGroups = Groups.Select(group => new SerializableWorkspaceSettingsGroup
318 {
319 GroupKey = group.GroupKey,
320 Settings = group.Settings.Select(s => new SerializableWorkspaceSetting
321 {
322 Key = s.Key,
323 Value = s.UntypedValue
324 }).ToList()
325 }).ToList();
326
327 fileService.Save(Path.GetDirectoryName(filePath), Path.GetFileName(filePath), serializableGroups);
328 }
329
337 public void Load(IFileService fileService, string filePath)
338 {
339 List<SerializableWorkspaceSettingsGroup> serializableGroups = fileService.Read<List<SerializableWorkspaceSettingsGroup>>(Path.GetDirectoryName(filePath), Path.GetFileName(filePath));
340 Groups.Clear();
341 initializeSettings();
342 if (serializableGroups != null)
343 {
344 foreach (SerializableWorkspaceSettingsGroup serializableGroup in serializableGroups)
345 {
346 foreach (SerializableWorkspaceSetting serializableSetting in serializableGroup.Settings)
347 {
348 IWorkspaceSetting setting = GetSetting(serializableGroup.GroupKey, serializableSetting.Key);
349 if (setting != null)
350 {
351 object serializableValue = serializableSetting.Value;
352 if(serializableValue is string strValue && setting.ValueType.IsEnum)
353 {
354 serializableValue = Enum.Parse(setting.ValueType, strValue);
355 }
356 setting.UntypedValue = serializableValue;
357 }
358 }
359 }
360 }
361 CreateSnapshot();
362 }
