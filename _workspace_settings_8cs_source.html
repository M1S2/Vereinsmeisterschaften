108 public void AddGroup(WorkspaceSettingsGroup group)
109 {
110 if (Groups.Where(g => string.Compare(g.GroupKey, group.GroupKey, true) == 0).Any())
111 {
112 // A group with the same key already exists
113 throw new ArgumentException($"A group with the key '{group.GroupKey}' already exists.", nameof(group));
114 }
115 else
116 {
117 Groups.Add(group);
118 group.PropertyChanged += (sender, e) =>
119 {
120 switch (e.PropertyName)
121 {
122 case nameof(WorkspaceSettingsGroup.HasChanged): OnPropertyChanged(nameof(HasChanged)); break;
123 default: break;
124 }
125 };
126 }
127 }

135 public WorkspaceSettingsGroup GetGroup(string groupKey, bool createGroupIfNotExisting = false)
136 {
137 WorkspaceSettingsGroup group = Groups.FirstOrDefault(g => string.Compare(g.GroupKey, groupKey, true) == 0);
138 if (group == null && createGroupIfNotExisting)
139 {
140 group = new WorkspaceSettingsGroup(groupKey);
141 AddGroup(group);
142 }
143 return group;
144 }

282 public void Save(IFileService fileService, string filePath)
283 {
284 CreateSnapshot();
285 List<SerializableWorkspaceSettingsGroup> serializableGroups = Groups.Select(group => new SerializableWorkspaceSettingsGroup

287 GroupKey = group.GroupKey,
288 Settings = group.Settings.Select(s => new SerializableWorkspaceSetting

293 }).ToList();
294
295 fileService.Save(Path.GetDirectoryName(filePath), Path.GetFileName(filePath), serializableGroups);
296 }

305 public void Load(IFileService fileService, string filePath)
306 {
307 List<SerializableWorkspaceSettingsGroup> serializableGroups = fileService.Read<List<SerializableWorkspaceSettingsGroup>>(Path.GetDirectoryName(filePath), Path.GetFileName(filePath));
308 Groups.Clear();
309 initializeSettings();
310 if (serializableGroups != null)
311 {
312 foreach (SerializableWorkspaceSettingsGroup serializableGroup in serializableGroups)
313 {
314 foreach (SerializableWorkspaceSetting serializableSetting in serializableGroup.Settings)
315 {
316 IWorkspaceSetting setting = GetSetting(serializableGroup.GroupKey, serializableSetting.Key);
317 if (setting != null)
318 {
319 object serializableValue = serializableSetting.Value;
320 if(serializableValue is string strValue && setting.ValueType.IsEnum)
321 {
322 serializableValue = Enum.Parse(setting.ValueType, strValue);
323 }
324 setting.UntypedValue = serializableValue;
325 }
326 }
327 }
328 }
329 CreateSnapshot();
330 }
