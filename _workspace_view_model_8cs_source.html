186 private async Task<(bool saveOut, bool cancelOut)> checkForUnsavedChangesAndQueryUserAction()
187 {
188 bool save = true, cancel = false;
189 if (_workspaceService?.HasUnsavedChanges ?? false)
190 {
191 MetroDialogSettings dialogButtonSettings = new MetroDialogSettings()
192 {
193 AffirmativeButtonText = Resources.SaveString,
194 NegativeButtonText = Resources.DontSaveString,
195 FirstAuxiliaryButtonText = Resources.CancelString,
196 DefaultButtonFocus = MessageDialogResult.Affirmative
197 };
198 MessageDialogResult dialogResult = await _dialogCoordinator.ShowMessageAsync(this, Resources.UnsavedChangesString, Resources.UnsavedChangesSavePromptString,
199 MessageDialogStyle.AffirmativeAndNegativeAndSingleAuxiliary, dialogButtonSettings);
200 switch (dialogResult)
201 {
202 case MessageDialogResult.Affirmative: save = true; cancel = false; break;
203 case MessageDialogResult.Negative: save = false; cancel = false; break;
204 case MessageDialogResult.FirstAuxiliary: save = false; cancel = true; break;
205 default: break;
206 }
207 }
208 return (save, cancel);
209 }

280 private void initSettingsGroups(WorkspaceSettings model)
281 {
282 if(model == null) { return; }
283 SettingsGroups = new ObservableCollection<WorkspaceSettingsGroupViewModel>();
284
285 ResourceDictionary settingEditorTemplatesResourceDictionary = new ResourceDictionary();
286 settingEditorTemplatesResourceDictionary.Source = new Uri("pack://application:,,,/Views/WorkspaceSettingEditorTemplates.xaml", UriKind.RelativeOrAbsolute);
287
288 foreach (KeyValuePair<string, string> groupViewConfig in WorkspaceSettingViewConfigs.GroupKeyLabelsDict)
289 {
290 // Get the group from the model. If no group with this key exists, skip it.
291 WorkspaceSettingsGroup groupModel = model?.Groups?.FirstOrDefault(g => g.GroupKey == groupViewConfig.Key);
292 if (groupModel == null) { continue; }
293
294 List<IWorkspaceSettingViewModel> settingsVms = new List<IWorkspaceSettingViewModel>();
295
296 // Iterate over the setting view configs
297 foreach (KeyValuePair<(string, string), WorkspaceSettingViewConfig> settingViewConfig in WorkspaceSettingViewConfigs.SettingKeyConfigDict)
298 {
299 (string groupKey, string settingKey) = settingViewConfig.Key;
300
301 // Only consider settings that belong to the current group
302 if (groupKey != groupViewConfig.Key) { continue; }
303
304 // Get the setting from the model. If no setting with this key exists, skip it.
305 IWorkspaceSetting settingModel = groupModel?.Settings?.FirstOrDefault(s => s.Key == settingKey);
306 if (settingModel == null) { continue; }
307
308 WorkspaceSettingEditorTypes editorType = settingViewConfig.Value?.Editor ?? WorkspaceSettingEditorTypes.String;
309 DataTemplate editorTemplate = settingEditorTemplatesResourceDictionary[editorType.ToString() + "EditorTemplate"] as DataTemplate;
310
311 // Create the view model for this setting
312 Type genericType = typeof(WorkspaceSettingViewModel<>).MakeGenericType(settingModel.ValueType);
313 IWorkspaceSettingViewModel settingVm = (IWorkspaceSettingViewModel)Activator.CreateInstance(genericType,
314 settingModel,
315 settingViewConfig.Value.Label,
316 settingViewConfig.Value.Tooltip,
317 settingViewConfig.Value.Icon,
318 settingViewConfig.Value.IconGeometry,
319 editorTemplate,
320 settingViewConfig.Value.SupportResetToDefault)!;
321 settingsVms.Add(settingVm);
322 }
323
324 WorkspaceSettingsGroupViewModel groupVm = new WorkspaceSettingsGroupViewModel(groupModel, groupViewConfig.Value, settingsVms);
325 SettingsGroups.Add(groupVm);
326 }
327 }

352 private void _workspaceService_PropertyChanged(object sender, System.ComponentModel.PropertyChangedEventArgs e)
353 {
354 switch (e.PropertyName)
355 {
356 case nameof(IWorkspaceService.HasUnsavedChanges): OnPropertyChanged(nameof(HasUnsavedChanges)); break;
357 case nameof(IWorkspaceService.HasUnsavedChanges_Persons): OnPropertyChanged(nameof(HasUnsavedChanges_Persons)); break;
358 case nameof(IWorkspaceService.HasUnsavedChanges_Competitions): OnPropertyChanged(nameof(HasUnsavedChanges_Competitions)); break;
359 case nameof(IWorkspaceService.HasUnsavedChanges_Races): OnPropertyChanged(nameof(HasUnsavedChanges_Races)); break;
360 case nameof(IWorkspaceService.HasUnsavedChanges_Settings):
361 {
362 OnPropertyChanged(nameof(HasUnsavedChanges_Settings));
363 OnPropertyChanged(nameof(SettingsGroups));
364 break;
365 }
366 case nameof(IWorkspaceService.PersistentPath): OnPropertyChanged(nameof(CurrentWorkspaceFolder)); break;
367 case nameof(IWorkspaceService.IsWorkspaceOpen):
368 {
369 OnPropertyChanged(nameof(CurrentWorkspaceFolder));
370 OnPropertyChanged(nameof(Settings));
371 ((RelayCommand)LoadWorkspaceCommand).NotifyCanExecuteChanged();
372 ((RelayCommand)SaveWorkspaceCommand).NotifyCanExecuteChanged();
373 ((RelayCommand)CloseWorkspaceCommand).NotifyCanExecuteChanged();
374 ((RelayCommand)OpenWorkspaceFolderCommand).NotifyCanExecuteChanged();
375 break;
376 }
377 default: break;
378 }
